131   private Map<String, String> reverseMap(Map<String, String> map) {
132     Map<String, String> reverse = new HashMap<String, String>();
133     for (Map.Entry<String, String> e : map.entrySet())
134       reverse.put(e.getValue(), e.getKey());
135     return reverse;
136   }

158   protected void parseExperimentalDesign() {
159     String id = ExperimentCodeFunctions.getInfoExperimentID(space, projectCode);
160     List<Experiment> exps = openbis.getExperimentById2(id);
161     if (!exps.isEmpty()) {
162       StudyXMLParser parser = new StudyXMLParser();
163       try {
164         JAXBElement<Qexperiment> design =
165             parser.parseXMLString(exps.get(0).getProperties().get("Q_EXPERIMENTAL_SETUP"));
166         factorLabels = parser.getFactorLabels(design);
167         factorsForLabelsAndSamples = parser.getFactorsForLabelsAndSamples(design);
168       } catch (JAXBException e) {
169         // TODO Auto-generated catch block
170         e.printStackTrace();
171       }
172     }

175   private VerticalLayout computePopupComponent() {
176     initDocx4J();
177     P p = docxHelper.createParagraph("Summary for project " + projectCode, true, false, "40");
178     wordMLPackage.getMainDocumentPart().addObject(p);
179 
180     VerticalLayout res = new VerticalLayout();
181     res.setCaption("Summary");
182     res.setSpacing(true);
183     res.setMargin(true);
184 
185     // collect and connect everything (if samples exist)
186     List<Sample> samples =
187         openbis.getSamplesWithParentsAndChildrenOfProjectBySearchService(projectCode);
188     if (!projectCode.startsWith("Q") || projectCode.length() != 5) {
189       success = false;
190       return res;
191     }
192     if (samples.size() == 0) {
193       success = false;
194       return res;
195     } else {
196       List<Experiment> experiments = openbis.getExperimentsForProject3(projectCode);
197       Experiment first = experiments.get(0);
198       List<DataSet> datasets = openbis.getDataSetsOfProjectByIdentifier(
199           first.getIdentifier().replace("/" + first.getCode(), ""));
200 
201       Map<Experiment, List<Sample>> expToSamples = new HashMap<Experiment, List<Sample>>();
202       Map<String, List<DataSet>> sampIDToDS = new HashMap<String, List<DataSet>>();
203       Map<String, List<DataSet>> expIDToDS = new HashMap<String, List<DataSet>>();
204       Map<String, Experiment> expIDToExp = new HashMap<String, Experiment>();
205       Map<String, List<PropertyType>> expTypeToProperties =
206           new HashMap<String, List<PropertyType>>();
207 
208       for (DataSet d : datasets) {
209         String sampID = d.getSampleIdentifierOrNull();
210         String expID = d.getExperimentIdentifier();
211         // fill sample id to datasets
212         if (sampIDToDS.containsKey(sampID))
213           sampIDToDS.get(sampID).add(d);
214         else
215           sampIDToDS.put(sampID, new ArrayList<DataSet>(Arrays.asList(d)));
216         // fill experiment id to datasets
217         if (expIDToDS.containsKey(expID))
218           expIDToDS.get(expID).add(d);
219         else
220           expIDToDS.put(expID, new ArrayList<DataSet>(Arrays.asList(d)));
221       }
222       for (Experiment e : experiments)
223         expIDToExp.put(e.getIdentifier(), e);
224       for (Sample s : samples) {
225         Experiment exp = expIDToExp.get(s.getExperimentIdentifierOrNull());
226         if (expToSamples.containsKey(exp))
227           expToSamples.get(exp).add(s);
228         else
229           expToSamples.put(exp, new ArrayList<Sample>(Arrays.asList(s)));
230       }
231       // sort and display information
232       Collections.sort(experiments, ExperimentTypeComparator.getInstance());
233       for (Experiment e : experiments) {
234 
235         String type = e.getExperimentTypeCode();
236         if (!expTypeToProperties.containsKey(type))
237           expTypeToProperties.put(type,
238               openbis.listPropertiesForType(openbis.getExperimentTypeByString(type)));
239         if (expTypeTranslation.containsKey(type))
240           type = expTypeTranslation.get(type);
241         if (expToSamples.containsKey(e) && !type.equals("Project Details")) {// TODO here, project
242                                                                              // details/attachments
243                                                                              // information could be
244                                                                              // added
245           VerticalLayout section = new VerticalLayout();
246           // create vaadin and docx section
247           generateExperimentHeaderWithMetadata(e, section, wordMLPackage.getMainDocumentPart(),
248               expTypeToProperties);
249 
250           Table sampleTable = generateSampleTable(expToSamples.get(e), sampIDToDS,
251               expIDToDS.get(e.getIdentifier()), wordMLPackage.getMainDocumentPart());
252           section.addComponent(sampleTable);
253           res.addComponent(section);
254         }
255       }
256       success = true;
257     }
258     addSummaryDownload(res);
259     return res;
260   }

292   private void generateExperimentHeaderWithMetadata(Experiment e, VerticalLayout section,
293       MainDocumentPart mainDocumentPart, Map<String, List<PropertyType>> expTypeToProperties) {
294     String expHeadline =
295         expTypeTranslation.get(e.getExperimentTypeCode()) + " (" + e.getCode() + ")";
296 
297     // docx
298     P p1 = docxHelper.createParagraph(expHeadline, true, false, "32");
299     mainDocumentPart.addObject(p1);
300 
301     // view
302     Panel expPanel = new Panel();
303     expPanel.setStyleName(ValoTheme.PANEL_WELL);
304     Label header = new Label(expHeadline);
305     header.setStyleName(ValoTheme.LABEL_BOLD);
306     section.addComponent(header);
307 
308     // all possible properties that can be set for this experiment type
309     List<PropertyType> possibleProps = expTypeToProperties.get(e.getExperimentTypeCode());
310     Map<String, String> properties = e.getProperties();
311     // collect ready-to-display key-value pairs
312     List<String> metadata = new ArrayList<String>();
313 
314     Date date = e.getRegistrationDetails().getRegistrationDate();
315     SimpleDateFormat dateFormat = new SimpleDateFormat("dd-MM-yy");
316     String formattedDate = dateFormat.format(date);
317     metadata.add("Date: " + formattedDate);
318     for (PropertyType p : possibleProps) {
319       String code = p.getCode();
320       if (properties.containsKey(code)) {
321         String label = p.getLabel();
322         String value = properties.get(code);
323         if (label.equals("Notes")) {
324           metadata.addAll(parseXMLNotes(value));
325         } else {
326           if (allMap.containsKey(value)) {
327             value = allMap.get(value);
328           }
329           metadata.add(label + ": " + value);
330         }
331       }
332     }
333 
334     Collections.sort(metadata);
335     for (String line : metadata) {
336       // vaadin
337       Label l = new Label(line);
338       section.addComponent(l);
339       // docx
340       mainDocumentPart.addObject(docxHelper.createParagraph(line, false, false, "32"));
341     }
342     expPanel.setContent(section);
343   }

345   private List<String> parseXMLNotes(String value) {
346     List<String> res = new ArrayList<String>();
347     List<Note> notes = new ArrayList<Note>();
348     try {
349       JAXBElement<Notes> jaxbelem = HistoryReader.parseNotes(value);
350       notes = jaxbelem.getValue().getNote();
351     } catch (JAXBException e) {
352       // TODO Auto-generated catch block
353       e.printStackTrace();
354     }
355     for (Note n : notes) {
356       res.add(Utils.usernameToFullName(n.getUsername()) + " commented: " + n.getComment());
357     }
358     return res;
359   }

550   private String parseTissue(Sample s) {
551     String tissue = s.getProperties().get("Q_PRIMARY_TISSUE");
552     String other = s.getProperties().get("Q_TISSUE_DETAILED");
553     if (tissue.equals("OTHER"))
554       if (other == null || other.isEmpty())
555         tissue = "Other";
556       else
557         tissue = other;
558     else if (tissue.equals("CELL_CULTURE"))
559       if (other == null || other.isEmpty())
560         tissue = "Cell Culture";
561       else
562         tissue = other;
563     if (allMap.containsKey(tissue))
564       return allMap.get(tissue);
565     else
566       return tissue;
567   }
