263   private void initTSVDownloads(String space, String project) {
264     tsvDownloadContent = new TSVDownloadComponent();
265     tsvDownloadContent.setVisible(false);
266     List<String> types = new ArrayList<String>(
267         Arrays.asList("Q_BIOLOGICAL_ENTITY", "Q_BIOLOGICAL_SAMPLE", "Q_TEST_SAMPLE"));
268     boolean tsvable = false;
269     for (Sample s : datahandler.getOpenBisClient()
270         .getSamplesOfProjectBySearchService("/" + space + "/" + project))
271       if (types.contains(s.getSampleTypeCode())) {
272         tsvable = true;
273         break;
274       }
275     if (tsvable) {
276       // need to be disabled first so old project tsvs are not downloadable
277       tsvDownloadContent.disableSpreadSheets();
278       tsvDownloadContent.prepareSpreadsheets(types, space, project, datahandler.getOpenBisClient(),
279           datahandler.getFactorLabels(), datahandler.getFactorsForLabelsAndSamples());
280       tsvDownloadContent.setVisible(true);
281     } else {
282       // nothing to create a tsv from
283       tsvDownloadContent.setVisible(false);
284     }
285   }

999   /**
1000    * The input should have the following form: type=openbis_type&id=openbis_id e.g.
1001    * type=sample&id=/ABI_SYSBIO/QMARI117AV It is specifically designed to be used in the case of
1002    * datasetView. In other cases there is no guarantee that it will work correctly. returns a map
1003    * with two entries: "type": "openbistype" "id" : "openbisId"
1004    * 
1005    * @param parameters
1006    * @return
1007    */
1008   public static Map<String, String> getMap(String parameters) {
1009     if (parameters == null || parameters.equals(""))
1010       return null;
1011     String[] params = parameters.split("&");
1012     // TODO check for length == 2 needed ?
1013     // if (params == null || params.length != 2)
1014     if (params == null || params.length > 3)
1015       return null;
1016     HashMap<String, String> map = new HashMap<String, String>();
1017     for (String p : params) {
1018       String[] kv = p.split("=");
1019       if (kv.length != 2)
1020         return null;
1021       map.put(kv[0], kv[1]);
1022     }
1023     return map;
