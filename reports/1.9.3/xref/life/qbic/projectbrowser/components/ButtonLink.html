18  package life.qbic.projectbrowser.components;
19  
20  import com.vaadin.server.ExternalResource;
21  import com.vaadin.shared.ui.label.ContentMode;
22  import com.vaadin.ui.Label;
23  
24  
25  /*
26   * Link that looks like a Button, that is actually a Label Based on
27   * https://vaadin.com/forum/#!/thread/69989
28   */
29  public class ButtonLink extends Label{
30    /**
31     * 
32     */
33    private static final long serialVersionUID = 4654807493467264366L;
34    String caption;
35  
36    /**
37     * Link that looks like a Button, that is actually a Label. Set the caption of the "Button" and
38     * the externalResource which should contain a url
39     * 
40     * @param caption
41     * @param externalResource
42     */
43    public ButtonLink(String caption, ExternalResource externalResource) {
44      super();
45      super.setWidth(null);
46      this.caption = caption;
47      setResource(externalResource);
48    }
49  
50    public void setResource(ExternalResource externalResource) {
51      buildHTMLCode(externalResource.getURL());
52      super.setContentMode(ContentMode.HTML);
53    }
54  
55    // The following lines are copy pasted from rendered Vaadin v6.1 buttons.
56    private void buildHTMLCode(String url) {
57      StringBuilder sb = new StringBuilder("<a href='");
58      sb.append(url);
59      sb.append("' style='text-decoration: display: block;'>");
60      sb.append("<div class='v-button' tabindex='0' style='width: 100%;'>");
61      sb.append("<span class='v-button-wrap'>");
62      sb.append(caption);
63      sb.append("</span>");
64      sb.append("</div>");
65      sb.append("</a>");
66      super.setValue(sb.toString());
67    }
68  
69  }
