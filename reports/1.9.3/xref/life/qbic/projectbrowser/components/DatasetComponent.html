126     public void updateUI(String type, String id) {
127 
128         if (id == null) {
129             return;
130         }
131         try {
132             HierarchicalContainer datasetContainer = new HierarchicalContainer();
133             datasetContainer.addContainerProperty("Select", CheckBox.class, null);
134             datasetContainer.addContainerProperty("Project", String.class, null);
135             datasetContainer.addContainerProperty("Sample", String.class, null);
136             datasetContainer.addContainerProperty("Description", String.class, null);
137             datasetContainer.addContainerProperty("File Name", String.class, null);
138             datasetContainer.addContainerProperty("File Type", String.class, null);
139             datasetContainer.addContainerProperty("Dataset Type", String.class, null);
140             datasetContainer.addContainerProperty("Registration Date", String.class, null);
141             datasetContainer.addContainerProperty("Validated", Boolean.class, null);
142             datasetContainer.addContainerProperty("File Size", String.class, null);
143             datasetContainer.addContainerProperty("file_size_bytes", Long.class, null);
144             datasetContainer.addContainerProperty("dl_link", String.class, null);
145             datasetContainer.addContainerProperty("isDirectory", Boolean.class, null);
146             datasetContainer.addContainerProperty("CODE", String.class, null);
147 
148             List<ch.systemsx.cisd.openbis.generic.shared.api.v1.dto.DataSet> retrievedDatasets = null;
149 
150             // clear download queue for new view
151             PortletSession portletSession = ((ProjectBrowserPortlet) UI.getCurrent()).getPortletSession();
152             portletSession.setAttribute("qbic_download",
153                 new HashMap<String, SimpleEntry<String, Long>>(),
154                 PortletSession.APPLICATION_SCOPE);
155 
156             Map<String, Sample> checkedTestSamples = new HashMap<String, Sample>();
157 
158             switch (type) {
159                 case "project":
160                     String projectIdentifier = id;
161                     retrievedDatasets = datahandler.getOpenBisClient()
162                         .getDataSetsOfProjectByIdentifierWithSearchCriteria(projectIdentifier);
163 
164                     List<Sample> allSamples = datahandler.getOpenBisClient()
165                         .getSamplesWithParentsAndChildrenOfProjectBySearchService(projectIdentifier);
166 
167                     for (Sample sample : allSamples) {
168                         checkedTestSamples.put(sample.getCode(), sample);
169                     }
170                     break;
171 
172                 case "experiment":
173                     String experimentIdentifier = id;
174                     retrievedDatasets = datahandler.getOpenBisClient()
175                         .getDataSetsOfExperimentByCodeWithSearchCriteria(experimentIdentifier);
176 
177                     Project proj = datahandler.getOpenBisClient()
178                         .getProjectOfExperimentByIdentifier(experimentIdentifier);
179 
180                     List<Sample> extSamples = datahandler.getOpenBisClient()
181                         .getSamplesWithParentsAndChildrenOfProjectBySearchService(proj.getIdentifier());
182 
183                     for (Sample sample : extSamples) {
184                         checkedTestSamples.put(sample.getCode(), sample);
185                     }
186                     break;
187 
188                 case "sample":
189                     String sampleIdentifier = id;
190                     retrievedDatasets =
191                         new ArrayList<ch.systemsx.cisd.openbis.generic.shared.api.v1.dto.DataSet>();
192 
193                     String code = sampleIdentifier.split("/")[2];
194                     Sample start =
195                         datahandler.getOpenBisClient().getSamplesWithParentsAndChildren(code).get(0);
196 
197                     Project sampProject = datahandler.getOpenBisClient()
198                         .getProjectOfExperimentByIdentifier(start.getExperimentIdentifierOrNull());
199 
200                     retrievedDatasets
201                         .addAll(datahandler.getOpenBisClient().getDataSetsOfSample(start.getCode()));
202 
203                     List<Sample> allProjSamples = datahandler.getOpenBisClient()
204                         .getSamplesWithParentsAndChildrenOfProjectBySearchService(
205                             sampProject.getIdentifier());
206 
207                     for (Sample sample : allProjSamples) {
208                         checkedTestSamples.put(sample.getCode(), sample);
209                     }
210 
211                     Set<Sample> startList = new HashSet<Sample>();
212                     Set<Sample> allChildren = getAllChildren(startList, start);
213 
214                     for (Sample samp : allChildren) {
215                         retrievedDatasets
216                             .addAll(datahandler.getOpenBisClient().getDataSetsOfSample(samp.getCode()));
217                     }
218                     break;
219 
220                 default:
221                     retrievedDatasets =
222                         new ArrayList<ch.systemsx.cisd.openbis.generic.shared.api.v1.dto.DataSet>();
223                     break;
224             }
225 
226             numberOfDatasets = retrievedDatasets.size();
227 
228             final BeanItemContainer<DatasetBean> forExport = new BeanItemContainer(DatasetBean.class);
229             if (numberOfDatasets == 0) {
230 
231                 if (type.equals("project")) {
232                     headerLabel.setValue(String.format(
233                         "This view shows all datasets associated with this project. There are %s registered datasets.",
234                         numberOfDatasets));
235 
236                     Utils.Notification("No datasets available.",
237                         "No data is available for this project. Please contact the project manager if this is not expected.",
238                         "warning");
239                 } else {
240                     headerLabel.setValue(String.format(
241                         "This view shows all datasets associated with this sample (including samples which have been derived from this sample). There are %s registered datasets.",
242                         numberOfDatasets));
243 
244                     Utils.Notification("No datasets available.",
245                         "No data is connected to this sample. Please contact the project manager if this is not expected.",
246                         "warning");
247                 }
248             } else {
249 
250                 Map<String, String> samples = new HashMap<String, String>();
251 
252                 // project same for all datasets
253                 String projectCode = retrievedDatasets.get(0).getExperimentIdentifier().split("/")[2];
254                 for (ch.systemsx.cisd.openbis.generic.shared.api.v1.dto.DataSet dataset : retrievedDatasets) {
255                     samples.put(dataset.getCode(), dataset.getSampleIdentifierOrNull().split("/")[2]);
256                 }
257 
258                 List<DatasetBean> dsBeans = datahandler.queryDatasetsForFolderStructure(retrievedDatasets);
259 
260                 numberOfDatasets = dsBeans.size();
261 
262                 if (type.equals("project")) {
263                     headerLabel.setValue(String.format(
264                         "This view shows all datasets associated with this project. There are %s registered datasets.",
265                         numberOfDatasets));
266                 } else if (type.equals("sample")) {
267                     headerLabel.setValue(String.format(
268                         "This view shows all datasets associated with this sample (including samples which have been derived from this sample). There are %s registered datasets.",
269                         numberOfDatasets));
270                 }
271 
272                 for (DatasetBean d : dsBeans) {
273                     Date date = d.getRegistrationDate();
274                     SimpleDateFormat sd = new SimpleDateFormat("yyyy-MM-dd hh:mm a");
275                     String dateString = sd.format(date);
276                     String sampleID = samples.get(d.getCode());
277 
278                     Sample dsSample = checkedTestSamples.get(sampleID);
279 
280                     String secNameDS = d.getProperties().get("Q_SECONDARY_NAME");
281 
282                     String secName = datahandler.getSecondaryName(dsSample, secNameDS);
283 
284                     forExport.addBean(d);
285 
286                     registerDatasetInTable(d, datasetContainer, projectCode, sampleID, dateString, null, secName);
287                 }
288             }
289 
290             this.setContainerDataSource(datasetContainer);
291             prepareTSVExportFile(forExport, id);
292         } catch (Exception e) {
293             LOG.error(String.format("getting dataset failed for dataset %s %s", type, id), e);
294         }
295     }

297     private synchronized void prepareTSVExportFile(final BeanItemContainer<DatasetBean> itemContainer, final String id) {
298         // disable tsv export button (it will be enabled by the background thread preparing the export)
299         tsvExportButton.setEnabled(false);
300         tsvExportButton.setDescription("Please wait a moment, your TSV export data is being prepared in the background.");
301         // remove the button from the downloader
302         if (fileDownloader != null) {
303             tsvExportButton.removeExtension(fileDownloader);
304         }
305         UI.getCurrent().setPollInterval(150);
306         CompletableFuture.supplyAsync(() -> Utils.containerToString(itemContainer))
307             .thenApplyAsync(tsvContent -> Utils.getTSVStream(tsvContent, String.format("%s_%s_", id.substring(1).replace("/", "_"), "registered_datasets")))
308             .thenAccept((streamResource) -> {
309                 // update UI
310                 UI.getCurrent().access(() -> {
311                     fileDownloader = new FileDownloader(streamResource);
312                     fileDownloader.extend(tsvExportButton);
313                     tsvExportButton.setEnabled(true);
314                     tsvExportButton.setDescription("Click to download");
315                 });
316                 UI.getCurrent().setPollInterval(-1);
317             });
318     }

585     private void setCheckedBox(Object itemId, String parentFolder) {
586         CheckBox itemCheckBox =
587             (CheckBox) this.table.getItem(itemId).getItemProperty("Select").getValue();
588         itemCheckBox.addValueChangeListener(new TableCheckBoxValueChangeListener(itemId, parentFolder));
589 
590         if (table.hasChildren(itemId)) {
591             for (Object childId : table.getChildren(itemId)) {
592                 String newParentFolder = Paths
593                     .get(parentFolder,
594                         (String) this.table.getItem(itemId).getItemProperty("File Name").getValue())
595                     .toString();
596                 setCheckedBox(childId, newParentFolder);
597             }
598         }

782     /**
783      * The input should have the following form: type=openbis_type&id=openbis_id e.g. type=sample&id=/ABI_SYSBIO/QMARI117AV It is specifically designed to be
784      * used in the case of datasetView. In other cases there is no guarantee that it will work correctly. returns a map with two entries: "type": "openbistype"
785      * "id" : "openbisId"
786      */
787     public static Map<String, String> getMap(String parameters) {
788         if (parameters == null || parameters.equals("")) {
789             return null;
790         }
791         String[] params = parameters.split("&");
792         // TODO check for length == 2 needed ?
793         // if (params == null || params.length != 2)
794         if (params == null || params.length > 3) {
795             return null;
796         }
797         HashMap<String, String> map = new HashMap<String, String>();
798         for (String p : params) {
799             String[] kv = p.split("=");
800             if (kv.length != 2) {
801                 return null;
802             }
803             map.put(kv[0], kv[1]);
804         }
805         return map;
806     }
