83    public void buildForm(final Workflow workFlow) {
84  
85      parameterForm.removeAllComponents();
86      parameterFieldGroup = new FieldGroup();
87      inputListFieldGroup = new FieldGroup();
88  
89      /*
90       * for (Map.Entry<String, Parameter> entry : workFlow.getData().getData().entrySet()) {
91       * FileParameter param = (FileParameter) entry.getValue(); FileNameValidator fileNameValidator =
92       * new FileNameValidator("Please provide a valid file path"); TextField newField =
93       * createInputField(param, fileNameValidator);
94       * 
95       * parameterForm.addComponent(newField); inputListFieldGroup.bind(newField, entry.getKey()); //
96       * Have to set it here because field gets cleared upon binding
97       * newField.setValue(param.getValue().toString()); }
98       */
99  
100     for (Map.Entry<String, Parameter> entry : workFlow.getParameters().getParams().entrySet()) {
101       if (entry.getValue() instanceof FloatParameter) {
102         FloatParameter param = (FloatParameter) entry.getValue();
103         FloatRangeValidator floatValidator =
104             new FloatRangeValidator(String.format("Parameter has to be in the range of %s to %s",
105                 param.getMinimum(), param.getMaximum()), param.getMinimum(), param.getMaximum());
106         TextField newField =
107             createParameterField(param, floatValidator, new StringToFloatConverter());
108 
109         parameterForm.addComponent(newField);
110         parameterFieldGroup.bind(newField, entry.getKey());
111         // Have to set it here because field gets cleared upon binding
112         newField.setValue(param.getValue().toString());
113         newField.setRequired(param.isRequired());
114       }
115 
116       else if (entry.getValue() instanceof IntParameter) {
117         IntParameter param = (IntParameter) entry.getValue();
118         IntegerRangeValidator intValidator =
119             new IntegerRangeValidator(String.format("Parameter has to be in the range of %s to %s",
120                 param.getMinimum(), param.getMaximum()), param.getMinimum(), param.getMaximum());
121         TextField newField =
122             createParameterField(param, intValidator, new StringToIntegerConverter());
123 
124         parameterForm.addComponent(newField);
125         parameterFieldGroup.bind(newField, entry.getKey());
126         // Have to set it here because field gets cleared upon binding
127         newField.setValue(param.getValue().toString());
128         newField.setRequired(param.isRequired());
129       }
130 
131       else if (entry.getValue() instanceof StringParameter) {
132         StringParameter param = (StringParameter) entry.getValue();
133         // necessary for Microarray QC to create ComboBox instead of TextField
134         if (param.getRange().size() == 0 && !workFlow.getName().equals("Microarray QC")) {
135           TextField newField = createInputField(param, null);
136 
137           parameterForm.addComponent(newField);
138           parameterFieldGroup.bind(newField, entry.getKey());
139           // Have to set it here because field gets cleared upon binding
140           newField.setValue(param.getValue().toString());
141           newField.setRequired(param.isRequired());
142         } else {
143           ComboBox newField = createStringSelectionParameterField(param);
144           parameterForm.addComponent(newField);
145           parameterFieldGroup.bind(newField, entry.getKey());
146           // Have to set it here because field gets cleared upon binding
147           newField.setValue(param.getValue().toString());
148           newField.setRequired(param.isRequired());
149         }
150       }
151 
152       else if (entry.getValue() instanceof BooleanParameter) {
153         BooleanParameter param = (BooleanParameter) entry.getValue();
154 
155         CheckBox newField = createParameterCheckBox(param);
156         newField.setValue((boolean) param.getValue());
157         parameterForm.addComponent(newField);
158         parameterFieldGroup.bind(newField, entry.getKey());
159         // Have to set it here because field gets cleared upon binding
160         newField.setValue((Boolean) param.getValue());
161         newField.setRequired(param.isRequired());
162       }
163     }

166   @Override
167   public Workflow getWorkflow() {
168     boolean parametersValid = writeSetParameters();
169     if (!parametersValid)
170       return null;
171     writetInputList();
172     return this.workFlow;
173   }

175   /**
176    * writes UI parameters to their model(workflow) equivalent returns true if all fields where set
177    * with meaningfull values.
178    * 
179    * @return
180    */
181   boolean writeSetParameters() {
182     Collection<Field<?>> registeredFields = parameterFieldGroup.getFields();
183     ParameterSet paramSet = workFlow.getParameters();
184 
185     for (Field<?> field : registeredFields) {
186       if (!field.isValid() || (field.isEmpty() & field.isRequired())) {
187         String errorMessage = "Warning: Parameter " + field.getDescription() + "is invalid!";
188         Utils
189             .Notification(
190                 "Invalid parameter setting.",
191                 "Value for parameter \""
192                     + field.getDescription()
193                     + String
194                         .format(
195                             "\" is invalid or parameter field of required parameter (%s) must not be empty.",
196                             field.getDescription()), "error");
197         LOG.info(errorMessage);
198         return false;
199       }
200 
201       String value = field.getValue().toString();
202       // paramSet.getParam(field.getCaption()).setValue(value);
203       paramSet.getParam(field.getDescription()).setValue(value);
204     }
205     return true;
206   }

208   /**
209    * Can be used to add Options to a Combobox that are only available at runtime (e.g. are created
210    * from openBIS metainformation
211    * 
212    * @param caption Caption of the field, normally wf name followed by port and parameter name, e.g.
213    *        "Microarray QC.1.f"
214    * @param params Set of options for this parameter
215    */
216   public void setComboboxOptions(String caption, Set<String> params) {
217     for (Field<?> field : parameterFieldGroup.getFields()) {
218       if (field.getDescription().equals(caption)) {
219         if (field instanceof ComboBox) {
220           ((ComboBox) field).addItems(params);
221         }
222       }
223     }
224   }

257   private TextField createParameterField(Parameter param, Validator validator, Converter converter) {
258     // TextField field = new TextField(param.getTitle());
259     // field.setDescription(param.getDescription());
260     String description;
261     if (param.getDescription().contains("#br#")) {
262       description = param.getDescription().split("#br#")[0];
263     } else {
264       description = param.getDescription();
265     }
266 
267     TextField field = new TextField(description);
268     field.setDescription(param.getTitle());
269     field.addValidator(validator);
270     field.setNullRepresentation("");
271     field.setNullSettingAllowed(true);
272     field.setImmediate(true);
273     field.setConverter(converter);
274     return field;
275   }

310   private ComboBox createStringSelectionParameterField(StringParameter param) {
311     // ComboBox box = new ComboBox(param.getTitle());
312     // box.setDescription(param.getDescription());
313     // for openMS inis
314     String description;
315     if (param.getDescription().contains("#br#")) {
316       description = param.getDescription().split("#br#")[0];
317     } else {
318       description = param.getDescription();
319     }
320     ComboBox box = new ComboBox(description);
321     box.setDescription(param.getTitle());
322     box.setFilteringMode(FilteringMode.CONTAINS);
323     box.addItems(param.getRange());
324     // should only be the range.
325     box.setNullSelectionAllowed(false);
326     box.setImmediate(true);
327     return box;
328   }

337   @Override
338   public ParameterSet getParameters() {
339     Collection<Field<?>> registeredFields = parameterFieldGroup.getFields();
340     ParameterSet paramSet = workFlow.getParameters();
341 
342     Map<String, Parameter> updatedParams = new HashMap<String, Parameter>();
343 
344     for (Field<?> field : registeredFields) {
345       // Parameter updatedParam = paramSet.getParam(field.getCaption());
346       Parameter updatedParam = paramSet.getParam(field.getDescription());
347       updatedParam.setValue(field.getValue());
348       // updatedParams.put(updatedParam.getTitle(), updatedParam);
349       updatedParams.put(updatedParam.getDescription(), updatedParam);
350     }
351 
352     ParameterSet updatedParamSet =
353         new ParameterSet(workFlow.getName(), workFlow.getDescription(), updatedParams);
354     return updatedParamSet;
355   }
