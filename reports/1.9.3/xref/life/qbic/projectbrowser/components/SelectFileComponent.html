18  package life.qbic.projectbrowser.components;
19  
20  import com.vaadin.data.util.BeanItemContainer;
21  import com.vaadin.server.FontAwesome;
22  import com.vaadin.shared.ui.MarginInfo;
23  import com.vaadin.ui.Button;
24  import com.vaadin.ui.CustomComponent;
25  import com.vaadin.ui.Grid;
26  import com.vaadin.ui.Grid.SelectionMode;
27  import com.vaadin.ui.GridLayout;
28  import com.vaadin.ui.Label;
29  import com.vaadin.ui.VerticalLayout;
30  import com.vaadin.ui.themes.ValoTheme;
31  
32  public class SelectFileComponent extends CustomComponent {
33    private static final long serialVersionUID = -8740268090810853563L;
34    private Button toLeft;
35    private Button toRight;
36    private Grid available;
37    private Grid selected;
38  
39    public SelectFileComponent(String mainCaption, String info, String sourceCaption,
40        String destinationCaption, BeanItemContainer<?> source, BeanItemContainer<?> destination) {
41      setCaption(mainCaption);
42  
43      VerticalLayout files = new VerticalLayout();
44      files.setSpacing(true);
45  
46      // info label
47      Label rawFilesInfo = new Label(info);
48      rawFilesInfo.addStyleName(ValoTheme.LABEL_COLORED);
49      files.addComponent(rawFilesInfo);
50      files.setWidth("100%");
51  
52      // available files in openbis
53      available = new Grid(source);
54      available.setCaption(sourceCaption);
55      available.setSelectionMode(SelectionMode.MULTI);
56  
57      // selected files for anaylsis
58      selected = new Grid(destination);
59  
60      if (mainCaption.equals("Raw files")) {
61        available.removeColumn("fullPath");
62        available.removeColumn("openbisCode");
63      }
64  
65      else if (mainCaption.equals("")) {
66        available.removeColumn("name");
67        available.removeColumn("path");
68        selected.removeColumn("name");
69        selected.removeColumn("path");
70      }
71  
72      selected.setCaption(destinationCaption);
73      selected.setSelectionMode(SelectionMode.MULTI);
74  
75  
76      for (Grid.Column col : available.getColumns()) {
77        col.setWidthUndefined();
78      }
79  
80      // selectedFiles.set
81      // buttons to add or remove files
82      VerticalLayout buttons = new VerticalLayout();
83      toLeft = new Button();
84      toLeft.setIcon(FontAwesome.ARROW_LEFT);
85  
86      toRight = new Button();
87      toRight.setIcon(FontAwesome.ARROW_RIGHT);
88      buttons.addComponent(toRight);
89      buttons.addComponent(toLeft);
90  
91      GridLayout grids = new GridLayout(3, 1);
92      grids.setWidth("100%");
93  
94      // grids.setSpacing(true);
95      grids.addComponent(available, 0, 0);
96  
97      available.setWidth("100%");
98      grids.addComponent(buttons, 1, 0);
99      grids.addComponent(selected, 2, 0);
100     grids.setColumnExpandRatio(0, 0.45f);
101     grids.setColumnExpandRatio(1, 0.029f);
102     grids.setColumnExpandRatio(2, 0.45f);
103     grids.setSpacing(false);
104     grids.setComponentAlignment(buttons, com.vaadin.ui.Alignment.MIDDLE_CENTER);
105 
106     selected.setWidth("100%");
107 
108     files.addComponent(grids);
109 
110     this.setCompositionRoot(files);
111     files.setMargin(new MarginInfo(true, true, true, false));
112     this.setWidth("100%");
113 
114   }
115 
116   /**
117    * returns the button that should move files from left to right. Basically add to destination
118    * 
119    * @return
120    */
121   public Button getToRightButton() {
122     return toRight;
123   }
124 
125   /**
126    * returns the button to should move files from right to left. Basically remove from destination
127    * 
128    * @return
129    */
130   public Button getToLeftButton() {
131     return toLeft;
132   }
133 
134   public Grid getSource() {
135     return available;
136   }
137 
138   public Grid getDestination() {
139     return selected;
140   }
141 
142   public Grid getSelected() {
143     return selected;
144   }
145 
146   public void setSelected(Grid selected) {
147     this.selected = selected;
148   }
149 
150 
151 }
