60    private String buildImagePath() {
61      StringBuilder pathBuilder = new StringBuilder();
62  
63      if (PortalUtils.isLiferayPortlet()) {
64        Properties prop = new Properties();
65        InputStream in = this.getClass().getClassLoader()
66            .getResourceAsStream("WEB-INF/liferay-plugin-package.properties");
67        try {
68          prop.load(in);
69          in.close();
70        } catch (IOException e1) {
71          // TODO Auto-generated catch block
72          e1.printStackTrace();
73        }
74        String portletName = prop.getProperty("name");
75  
76        URI location = UI.getCurrent().getPage().getLocation();
77        // http
78        pathBuilder.append(location.getScheme());
79        pathBuilder.append("://");
80        // host+port
81        pathBuilder.append(location.getAuthority());
82  
83        String port = (Integer.toString(location.getPort()));
84        if (location.toString().contains(port)) {
85          pathBuilder.append(":");
86          pathBuilder.append(port);
87        }
88        pathBuilder.append("/");
89        pathBuilder.append(portletName);
90      }
91      pathBuilder.append("/VAADIN/img/");
92      return pathBuilder.toString();
93    }

95    public void loadProjectGraph(String projectIdentifier, List<Sample> samples,
96        List<DataSet> datasets, Set<String> factorLabels,
97        Map<Pair<String, String>, Property> factorsForLabelsAndSamples) {
98      this.factorBox = new ComboBox("Experimental Factor");
99      factorBox.setVisible(false);
100     addComponent(factorBox);
101 
102     currentSamples = samples;
103     if (currentSamples.isEmpty()) {
104       LOG.info("No samples to show found in this project.");
105     } else {
106       try {
107         // load here
108         structure = parser.parseSamplesBreadthFirst(currentSamples, datasets, factorLabels, factorsForLabelsAndSamples);
109         if (!structure.getFactorsToSamples().isEmpty()) {
110           factorBox.addItems(structure.getFactorsToSamples().keySet());
111           factorBox.setVisible(true);
112         }
113       } catch (JAXBException e) {
114         e.printStackTrace();
115         LOG.error("JAXB Parsing error!");
116       }
117     }
118 
119     factorBox.setImmediate(true);
120     factorBox.setNullSelectionAllowed(false);
121     final GraphPage parent = this;
122     factorBox.addValueChangeListener(new ValueChangeListener() {
123 
124       @Override
125       public void valueChange(ValueChangeEvent event) {
126         Object factor = factorBox.getValue();
127         if (sampleGraph != null) {
128           parent.removeComponent(sampleGraph);
129         }
130         sampleGraph = new ProjectGraph(parent, buildImagePath());
131         sampleGraph.setSizeFull();
132         parent.addComponent(sampleGraph);
133         if (factor != null) {
134           sampleGraph.setProject(structure.getFactorsToSamples().get(factor));
135         }
136       }
137     });
138   }
