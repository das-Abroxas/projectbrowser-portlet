29  public class SampleToBarcodeFieldTranslator {
30  
31    private static final Logger LOG = LogManager.getLogger(SampleToBarcodeFieldTranslator.class);
32  
33    public String buildInfo(ComboBox select, Sample s, String parents, boolean cut) {
34      Map<String, String> map = s.getProperties();
35      String in = "";
36      if (select.getValue() != null)
37        in = select.getValue().toString();
38      String res = "";
39      switch (in) {
40        case "Tissue/Extr. Material":
41          if (map.containsKey("Q_PRIMARY_TISSUE"))
42            res = map.get("Q_PRIMARY_TISSUE");
43          else
44            res = map.get("Q_SAMPLE_TYPE");
45          break;
46        case "Secondary Name":
47          res = map.get("Q_SECONDARY_NAME");
48          break;
49        case "Parent Samples (Source)":
50          if (parents == null) {
51            LOG
52                .error("Calls from BarcodePreviewComponent should not be able to select Parent samples as Info field choice."
53                    + "Setting from null to empty String to continue.");
54            parents = "";
55          }
56          res = parents;
57          break;
58        case "QBiC ID":
59          res = s.getCode();
60          break;
61        case "Lab ID":
62          res = map.get("Q_EXTERNALDB_ID");
63          break;
64        default:
65          if (!in.equals(""))
66            LOG.error("Unknown input: " + in + ". Field will be empty!");
67      }
68      if (res == null)
69        return "";
70      if (cut)
71        res = res.substring(0, Math.min(res.length(), 22));
72      return res;
73    }
74  
75    public String getCodeString(Sample sample, String codedName) {
76      Map<String, String> map = sample.getProperties();
77      String res = "";
78      // @SuppressWarnings("unchecked")
79      // Set<String> selection = (Set<String>) codedName.getValue();
80      // for (String s : selection) {
81      if (!res.isEmpty())
82        res += "_";
83      switch (codedName) {
84        case "QBiC ID":
85          res += sample.getCode();
86          break;
87        case "Secondary Name":
88          res += map.get("Q_SECONDARY_NAME");
89          break;
90        case "Lab ID":
91          res += map.get("Q_EXTERNALDB_ID");
92          break;
93      }
94      // }
95      res = fixFileName(res);
96      return res.substring(0, Math.min(res.length(), 21));
97    }
98  
99    private String fixFileName(String res) {
100     res = res.replace("null", "");
101     res = res.replace(";", "_");
102     res = res.replace("#", "_");
103     res = res.replace(" ", "_");
104     while (res.contains("__"))
105       res = res.replace("__", "_");
106     return res;
107   }
108 }
