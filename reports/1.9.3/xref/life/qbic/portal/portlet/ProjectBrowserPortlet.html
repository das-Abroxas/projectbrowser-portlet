86    @Override
87    protected Layout getPortletContent(final VaadinRequest request) {
88      LOG.info("Generating content for {}", ProjectBrowserPortlet.class);
89  
90      manager = ConfigurationManagerFactory.getInstance();
91      // check if we are allowed to display content to unauthenticated users
92      if (PortalUtils.isLiferayPortlet() && PortalUtils.getUser() == null) {
93        final ConfigurationManager configurationManager = ConfigurationManagerFactory.getInstance();
94        if (!configurationManager.isAllowUnauthenticatedAccess("projectbrowser")) {
95          // display a "please log in" message
96          return PortalUtils.buildNotLoggedInLayout();
97        }
98      }
99  
100     Layout mainLayout;
101 
102     // log who is connecting, when.
103     LOG.info(String.format("ProjectBrowser used by: %s", PortalUtils.getNonNullScreenName()));
104     // try to init connection to openbis and write some session attributes, that can be accessed globally
105     try {
106       initConnection();
107       initSessionAttributes();
108     } catch (Exception e) {
109       // probably the connection to openbis failed
110       buildOpenbisConnectionErrorLayout(request);
111       // write an error message if failed to load openbis and is in production
112       errorMessageIfIsProduction();
113     }
114     this.resUrl = (String) getPortletSession().getAttribute("resURL", PortletSession.APPLICATION_SCOPE);
115     mainLayout = initProgressBarAndThreading(request);
116 
117     return mainLayout;
118   }

120   /**
121    *
122    */
123   void errorMessageIfIsProduction() {
124     if (isInProductionMode())
125       try {
126         VaadinService.getCurrentResponse().sendError(HttpServletResponse.SC_GATEWAY_TIMEOUT,
127             "openbis could not be accessed.");
128       } catch (IOException | IllegalArgumentException e1) {
129         VaadinService.getCurrentResponse().setStatus(HttpServletResponse.SC_GATEWAY_TIMEOUT);
130       }
131   }

153   /**
154    * standard error layout, if openbis threw error on initialization despite a successful login
155    *
156    * @param request
157    */
158   private Layout buildUserUnknownError(final VaadinRequest request) {
159     VerticalLayout vl = new VerticalLayout();
160     setContent(vl);
161     vl.addComponent(new Label(
162         "An error occured while trying to load projects. Please contact your project manager to make sure your account is added to your projects."));
163     LOG.error(
164         "Couldn't initialize view. User is probably not added to openBIS and has been informed to contact project manager.");
165 
166     return vl;
167   }

169   /**
170    * starts the querying of openbis and initializing the view
171    *
172    * @param request
173    */
174   protected Layout initProgressBarAndThreading(VaadinRequest request) {
175     GridLayout layout = new GridLayout();
176 
177     try {
178       layout = buildMainLayout(datahandler, request, PortalUtils.getNonNullScreenName());
179     } catch (Exception e) {
180       if (datahandler.getOpenBisClient().loggedin()) {
181         LOG.error("User not known?", e);
182         buildUserUnknownError(request);
183       } else {
184         LOG.error("exception thrown during initialization.", e);
185       }
186     }
187 
188     return layout;
189   }
