159   public void update(Map<String, String> map) {
160 
161     if (map == null)
162       return;
163     // TODO In background thread?
164     type = map.get("type");
165     id = map.get("id");
166 
167     switch (type) {
168       case "project":
169         datasetBeans = controller.getcontainer(type, id);
170         List<String> datasetTypesInProject = new ArrayList<String>();
171 
172         for (Iterator<DatasetBean> i = datasetBeans.getItemIds().iterator(); i.hasNext();) {
173           DatasetBean dsBean = (DatasetBean) i.next();
174           datasetTypesInProject.add(dsBean.getFileType());
175         }
176         updateWorkflowSelection(datasetTypesInProject);
177         break;
178 
179       case "experiment":
180         break;
181 
182       case "sample":
183         break;
184 
185       case "workflowExperimentType":
186         String experiment = map.get("experiment");
187 
188         BeanItemContainer<Workflow> suitableWorkflows =
189             controller.suitableWorkflowsByExperimentType(experiment);
190 
191         BeanItemContainer<DatasetBean> suitableDatasets =
192             new BeanItemContainer<DatasetBean>(DatasetBean.class);
193 
194         List<String> workflowDatasetTypes = new ArrayList<String>();
195         for (Iterator i = suitableWorkflows.getItemIds().iterator(); i.hasNext();) {
196           Workflow workflowBean = (Workflow) i.next();
197 
198           Map<String, List<String>> requiredFiles = workflowBean.getFileTypes().get("required");
199 
200           for (Entry<String, List<String>> entry : requiredFiles.entrySet()) {
201             workflowDatasetTypes.addAll(entry.getValue());
202           }
203 
204         }
205 
206         for (Iterator i = controller.getcontainer("project", id).getItemIds().iterator(); i
207             .hasNext();) {
208           DatasetBean datasetBean = (DatasetBean) i.next();
209 
210           if (workflowDatasetTypes.contains(datasetBean.getFileType())) {
211             suitableDatasets.addBean(datasetBean);
212           }
213         }
214 
215         datasetBeans = suitableDatasets;
216         updateSelection(suitableWorkflows);
217         break;
218 
219       default:
220         updateSelection(new BeanItemContainer<Workflow>(Workflow.class));
221         break;
222     }
223   }

295     this.submission.setCaption(SUBMISSION_CAPTION + ": " + workFlow.getName());
296     this.submission.removeAllComponents();
297 
298     if (workFlow.getName().equals("MaxQuant")) {
299       BeanItemContainer<RawFilesBean> rawFilesBeans =
300           new BeanItemContainer<RawFilesBean>(RawFilesBean.class);
301       BeanItemContainer<FastaBean> selectedfastas =
302           new BeanItemContainer<FastaBean>(FastaBean.class);
303       BeanItemContainer<FastaBean> fastas = new BeanItemContainer<FastaBean>(FastaBean.class);
304       FastaDB db = new FastaDB();
305       fastas.addAll(db.getAll());
306 
307       Map<String, Parameter> params = workFlow.getData().getData();
308       BeanItemContainer<DatasetBean> subContainer =
309           new BeanItemContainer<DatasetBean>(DatasetBean.class);
310 
311 
312       for (Entry<String, Parameter> entry : params.entrySet()) {
313 
314         if (entry.getValue() instanceof FileParameter
315             || entry.getValue() instanceof FileListParameter) {
316 
317           List<String> associatedDataTypes = new ArrayList<String>();
318 
319           if (entry.getValue() instanceof FileParameter) {
320             associatedDataTypes = ((FileParameter) entry.getValue()).getRange();
321           } else if (entry.getValue() instanceof FileListParameter) {
322             associatedDataTypes = ((FileListParameter) entry.getValue()).getRange();
323           }
324 
325           if (associatedDataTypes.contains("fasta") || associatedDataTypes.contains("gtf")) {
326             continue;
327           } else {
328             for (Iterator<DatasetBean> i = projectDatasets.getItemIds().iterator(); i
329                 .hasNext();) {
330               DatasetBean dataset = i.next();
331 
332               if (associatedDataTypes.contains(dataset.getFileType())) {
333                 subContainer.addBean(dataset);
334 
335               }
336             }
337           }
338         }
339       }

619   /**
620    * submits workflow with given datasets
621    * 
622    * @throws SubmitFailedException
623    * @throws IllegalArgumentException
624    * @throws ConnectException
625    */
626   void submit(Workflow submittedWf, List<DatasetBean> selectedDatasets) throws ConnectException,
627       IllegalArgumentException, SubmitFailedException {
628     if (submittedWf == null || selectedDatasets.isEmpty()) {
629       return;
630     }
631     // THIS IS THE IMPORTANT LINE IN THAT MESS
632     String openbisId = controller.submitAndRegisterWf(type, id, submittedWf, selectedDatasets);
633     Utils.Notification("Workflow submitted",
634         "Workflow submitted successfully and saved under " + openbisId, "success");
635     // showNotification("Workflow submitted and saved under " + openbisId);
636   }

638   /**
639    * logs error and shows user that submission failed. Different exceptions yield different error
640    * messages
641    * 
642    * @param e
643    */
644   public void handleException(Exception e) {
645     if (e instanceof ConnectException || e instanceof IllegalArgumentException
646         || e instanceof SubmitFailedException) {
647       LOG.error("Submission failed, probably gUSE. " + e.getMessage(), e.getStackTrace());
648       Utils
649           .Notification(
650               "Workflow submission failed",
651               "Requested workflow could not be submitted due to internal errors. Please try again later and contact your project manager if the problem persists.",
652               "error");
653       try {
654         VaadinService
655             .getCurrentResponse()
656             .sendError(
657                 HttpServletResponse.SC_GATEWAY_TIMEOUT,
658                 "An error occured, while trying to connect to the database. Please try again later, or contact your project manager.");
659       } catch (IOException | IllegalArgumentException e1) {
660         // TODO Auto-generated catch block
661         LOG.error("Something went wrong: " + e.getMessage(), e.getStackTrace());
662         VaadinService.getCurrentResponse().setStatus(HttpServletResponse.SC_GATEWAY_TIMEOUT);
663       }
664     } else if (e instanceof RemoteAccessException) {
665       LOG.error("Submission failed, probably openbis. error message: " + e.getMessage(),
666           e.getStackTrace());
667       Utils
668           .Notification(
669               "Workflow submission failed",
670               "Requested workflow could not be submitted due to internal errors. Please try again later and contact your project manager if the problem persists.",
671               "error");
672     } else {
673       LOG.error("Internal error: " + e.getMessage(), e.getStackTrace());
674       Utils
675           .Notification(
676               "Workflow submission failed",
677               "Requested workflow could not be submitted due to internal errors. Please try again later and contact your project manager if the problem persists.",
678               "error");
679     }
