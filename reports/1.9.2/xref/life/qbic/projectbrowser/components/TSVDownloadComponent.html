97    public void prepareSpreadsheets(final List<String> sampleTypes, String space,
98        final String project, final OpenBisClient openbis, final Set<String> factors,
99        final Map<Pair<String, String>, Property> factorsForLabelsAndSamples) {
100     final TSVDownloadComponent layout = this;
101 
102     final Thread t = new Thread(() -> {
103       Map<String, String> tables = new HashMap<String, String>();
104       for (String type : sampleTypes) {
105         tables.put(type, getTSVString(openbis.getProjectTSV(project, type), factors,
106             factorsForLabelsAndSamples));
107       }
108       UI.getCurrent().access(new TSVReadyRunnable(layout, tables, project));
109       UI.getCurrent().setPollInterval(-1);
110     });
111     t.start();
112     UI.getCurrent().setPollInterval(100);
113   }

231   public void armButtons(List<StreamResource> streams) {
232     armDownloadButton(dlEntities, streams.get(0), 1);
233     armDownloadButton(dlExtracts, streams.get(1), 2);
234     if (streams.size() > 2)
235       armDownloadButton(dlPreps, streams.get(2), 3);
236   }
237 
238   protected void armDownloadButton(Button b, StreamResource stream, int dlnum) {
239     if (downloaders.size() < dlnum) {
240       FileDownloader dl = new FileDownloader(stream);
241       dl.extend(b);
242       downloaders.add(dl);
243     } else
244       downloaders.get(dlnum - 1).setFileDownloadResource(stream);
245     b.setEnabled(true);
246   }
247 
248   public void enableDownloads(boolean b) {
249     downloads.setEnabled(b);
250   }
