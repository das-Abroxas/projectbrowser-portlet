45  public class ChangeProjectMetadataComponent extends CustomComponent {
46    /**
47     * TODO generic component for experiments and samples
48     */
49    private static final long serialVersionUID = -5318223225284123020L;
50  
51    private static final Logger LOG = LogManager.getLogger(ChangeProjectMetadataComponent.class);
52  
53    private DataHandler datahandler;
54    private String resourceUrl;
55    private State state;
56    private VerticalLayout properties;
57  
58    private FormLayout form;
59    private FieldGroup fieldGroup;
60    VerticalLayout vert;
61  
62    private String currentDescription;
63  
64    public ChangeProjectMetadataComponent(DataHandler dh, State state, String resourceurl) {
65      this.datahandler = dh;
66      this.resourceUrl = resourceurl;
67      this.state = state;
68  
69      // this.setCaption("Metadata");
70  
71      this.initUI();
72    }
73  
74    private void initUI() {
75      properties = new VerticalLayout();
76      properties.setWidth(100.0f, Unit.PERCENTAGE);
77      properties.setMargin(new MarginInfo(true, false, true, true));
78      properties.setSpacing(true);
79  
80      this.setWidth(Page.getCurrent().getBrowserWindowWidth() * 0.8f, Unit.PIXELS);
81      this.setCompositionRoot(properties);
82    }
83  
84    public void updateUI(final ProjectBean projectBean) {
85      properties.removeAllComponents();
86      Button saveButton = new Button("Submit Changes");
87      saveButton.setStyleName(ValoTheme.BUTTON_FRIENDLY);
88  
89      currentDescription = projectBean.getDescription();
90  
91      saveButton.addClickListener(new ClickListener() {
92        @Override
93        public void buttonClick(final ClickEvent event) {
94          HashMap<String, Object> parameters = new HashMap<String, Object>();
95          Collection<Field<?>> registeredFields = fieldGroup.getFields();
96  
97          // List<Property> factors = new ArrayList<Property>();
98  
99          for (Field<?> field : registeredFields) {
100           parameters.put("description", field.getValue());
101         }
102 
103         parameters.put("identifier", projectBean.getId());
104         parameters.put("user", PortalUtils.getNonNullScreenName());
105         datahandler.getOpenBisClient().triggerIngestionService("update-project-metadata",
106             parameters);
107         Utils.Notification("Project details changed succesfully", String
108             .format("Details of project %s have been commited successfully.", projectBean.getId()),
109             "success");
110       }
111     });
112 
113     buildFormLayout();
114     Label desc = new Label(String.format(
115         "This view shows project details and can be used to change the corresponding values. \nIdentifier: %s",
116         projectBean.getId()), Label.CONTENT_PREFORMATTED);
117     desc.setWidth("50%");
118     properties.addComponent(desc);
119 
120     properties.addComponent(this.form);
121     properties.addComponent(saveButton);
122   }
123 
124 
125   private void buildFormLayout() {
126     final FieldGroup fieldGroup = new FieldGroup();
127     final FormLayout form2 = new FormLayout();
128 
129     TextArea tf = new TextArea("Description");
130     tf.setWidth("50%");
131     tf.setHeight("50%");
132     fieldGroup.bind(tf, "Description");
133     tf.setCaption("Description");
134     tf.setDescription("Description of this project.");
135     tf.setValue(currentDescription);
136     form2.addComponent(tf);
137 
138     this.fieldGroup = fieldGroup;
139     this.form = form2;
140   }
