18  package life.qbic.projectbrowser.components;
19  
20  
21  import com.vaadin.ui.CustomComponent;
22  import com.vaadin.ui.HorizontalLayout;
23  import com.vaadin.ui.OptionGroup;
24  
25  public class QuantificationLabelComponent extends CustomComponent{
26    private static final long serialVersionUID = -8953869755178358731L;
27    private HorizontalLayout mainlayout;
28    private OptionGroup light;
29    private OptionGroup medium;
30    private OptionGroup heavy;
31  
32    public QuantificationLabelComponent(){
33      mainlayout = new HorizontalLayout();
34      mainlayout.setCaption("Labels");
35      mainlayout.setSpacing(true);
36      light = labels("light labels");
37      medium = labels("medium labels");
38      heavy = labels("heavy labels");
39      setCompositionRoot(mainlayout);
40    }
41    
42    public void noLables() {
43      mainlayout.setVisible(false);
44    }
45  
46    public void lightAndHeavyLabels() {
47      mainlayout.setVisible(true);
48      mainlayout.removeAllComponents();
49      mainlayout.addComponent(light);
50      mainlayout.addComponent(heavy);
51    }
52  
53    public void lightMediumAndHeavyLabels() {
54      mainlayout.setVisible(true);
55      mainlayout.removeAllComponents();
56      mainlayout.addComponent(light);
57      mainlayout.addComponent(medium);
58      mainlayout.addComponent(heavy);
59      
60    }
61    OptionGroup labels(String caption){
62      OptionGroup labels = new OptionGroup(caption);
63      labels.setMultiSelect(true);
64      labels.addItems("Arg6", "Arg10", "Leu7", "Averigin");
65      return labels;  
66    }
67    OptionGroup getLightLabels(){
68      return light;
69    }
70    OptionGroup getMediumLabels(){
71      return medium;
72    }
73    OptionGroup getHeavyLabels(){
74      return heavy;
75    }
76    
77    
78  }
