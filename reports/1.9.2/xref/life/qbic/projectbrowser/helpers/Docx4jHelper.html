99    private void addTableCell(Tr tableRow, String content, int width,
100 
101       DocxStyle style, int horizontalMergedCells, String verticalMergedVal) {
102     Tc tableCell = factory.createTc();
103     addCellStyle(tableCell, content, style);
104     setCellWidth(tableCell, width);
105     setCellVMerge(tableCell, verticalMergedVal);
106     setCellHMerge(tableCell, horizontalMergedCells);
107     if (style.isNoWrap()) {
108       setCellNoWrap(tableCell);
109     }
110     tableRow.getContent().add(tableCell);
111   }

113   public P createParagraph(String content, boolean bold, boolean underlined, String fontSize) {
114     P paragraph = factory.createP();
115 
116     Text text = factory.createText();
117     text.setValue(content);
118 
119     R run = factory.createR();
120     run.getContent().add(text);
121 
122     paragraph.getContent().add(run);
123 
124     RPr runProperties = factory.createRPr();
125 
126     if (bold)
127       addBoldStyle(runProperties);
128     if (underlined)
129       addUnderlineStyle(runProperties);
130 
131     setFontSize(runProperties, fontSize);
132     run.setRPr(runProperties);
133 
134     return paragraph;
135   }

137   private void addCellStyle(Tc tableCell, String content, DocxStyle style) {
138     if (style != null) {
139 
140       P paragraph = factory.createP();
141 
142       Text text = factory.createText();
143       text.setValue(content);
144 
145       R run = factory.createR();
146       run.getContent().add(text);
147 
148       paragraph.getContent().add(run);
149 
150       setHorizontalAlignment(paragraph, style.getHorizAlignment());
151 
152       RPr runProperties = factory.createRPr();
153 
154       if (style.isBold()) {
155         addBoldStyle(runProperties);
156       }
157       if (style.isItalic()) {
158         addItalicStyle(runProperties);
159       }
160       if (style.isUnderline()) {
161         addUnderlineStyle(runProperties);
162       }
163 
164       setFontSize(runProperties, style.getFontSize());
165       setFontColor(runProperties, style.getFontColor());
166       setFontFamily(runProperties, style.getFontFamily());
167 
168       setCellMargins(tableCell, style.getTop(), style.getRight(),
169 
170           style.getBottom(), style.getLeft());
171       setCellColor(tableCell, style.getBackground());
172       setVerticalAlignment(tableCell, style.getVerticalAlignment());
173 
174       setCellBorders(tableCell, style.isBorderTop(), style.isBorderRight(),
175 
176           style.isBorderBottom(), style.isBorderLeft());
177 
178       run.setRPr(runProperties);
179 
180       tableCell.getContent().add(paragraph);
181     }
182   }

184   private void setCellBorders(Tc tableCell, boolean borderTop, boolean borderRight,
185 
186       boolean borderBottom, boolean borderLeft) {
187 
188     TcPr tableCellProperties = tableCell.getTcPr();
189     if (tableCellProperties == null) {
190       tableCellProperties = new TcPr();
191       tableCell.setTcPr(tableCellProperties);
192     }
193 
194     CTBorder border = new CTBorder();
195     // border.setColor("auto");
196     border.setColor("0000FF");
197     border.setSz(new BigInteger("20"));
198     border.setSpace(new BigInteger("0"));
199     border.setVal(STBorder.SINGLE);
200 
201     TcBorders borders = new TcBorders();
202     if (borderBottom) {
203       borders.setBottom(border);
204     }
205     if (borderTop) {
206       borders.setTop(border);
207     }
208     if (borderLeft) {
209       borders.setLeft(border);
210     }
211     if (borderRight) {
212       borders.setRight(border);
213     }
214     tableCellProperties.setTcBorders(borders);
215   }

299     }

302       TblWidth tW = new TblWidth();
303       tW.setType("dxa");
304       tW.setW(BigInteger.valueOf(top));
305       margins.setTop(tW);
306     }
307

310       lW.setType("dxa");

312       margins.setLeft(lW);
313     }

316       TblWidth rW = new TblWidth();
317       rW.setType("dxa");
318       rW.setW(BigInteger.valueOf(right));
319       margins.setRight(rW);
