256   public String getPrettyName(String uglyName) {
257     String prettyName = uglyName;
258 
259     if (namesMapping.containsKey(uglyName)) {
260       prettyName = namesMapping.get(uglyName);
261     }
262     return prettyName;
263   }
264 
265   public String getOpenBisName(String prettyName) {
266     String uglyName = prettyName;
267 
268     if (namesMapping.containsValue(prettyName)) {
269       uglyName = namesMapping.getKey(prettyName);
270     }
271     return uglyName;
272   }
