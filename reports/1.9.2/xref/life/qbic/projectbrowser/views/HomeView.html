350   @Override
351   public void enter(ViewChangeEvent event) {
352     try {
353       loadProjects();
354       int height = event.getNavigator().getUI().getPage().getBrowserWindowHeight();
355       int width = event.getNavigator().getUI().getPage().getBrowserWindowWidth();
356       buildLayout(height, width, event.getNavigator().getUI().getPage().getWebBrowser());
357     } catch (Exception e) {
358       LOG.error(String.format("failed to load projects for user %s", user), e);
359       removeAllComponents();
360       Label error = new Label("Connection to database interrupted. Please try again later.");
361       error.addStyleName(ValoTheme.LABEL_FAILURE);
362       error.addStyleName(ValoTheme.LABEL_HUGE);
363       addComponent(error);
364       setComponentAlignment(error, Alignment.MIDDLE_CENTER);
365     }
366   }

379   /**
380    * refresh all openbis project for current user. Basically currentBean is overwritten
381    */
382   public void loadProjects() {
383     // this.includePatientCreation = false;
384     final SpaceBean homeSpaceBean =
385         new SpaceBean("homeSpace", "", false, null, null, null, null, null, null);
386     BeanItemContainer<ProjectBean> projectContainer =
387         new BeanItemContainer<ProjectBean>(ProjectBean.class);
388 
389     LOG.info("Loading projects...");
390     List<Project> projects = datahandler.getOpenBisClient().getOpenbisInfoService()
391         .listProjectsOnBehalfOfUser(datahandler.getOpenBisClient().getSessionToken(), user);
392     LOG.info("Loading projects...done.");
393 
394     for (Project project : projects) {
395       String projectIdentifier = project.getIdentifier();
396       String projectCode = project.getCode();
397       String desc = project.getDescription();
398       if (desc == null) {
399         desc = "";
400       } else if (desc.length() > 0) {
401         desc = desc.substring(0, Math.min(desc.length(), 100));
402         if (desc.length() == 100) {
403           desc += "...";
404         }
405       }
406 
407       // TODO isn't this slow in this fashion? what about SELECT * and creating a map?
408       String secondaryName = datahandler.getDatabaseManager().getProjectName(projectIdentifier);
409       if (secondaryName.isEmpty() || secondaryName == null)
410         secondaryName = "n/a";
411 
412       ProjectBeann.html#ProjectBean">ProjectBean newProjectBean = new ProjectBean(projectIdentifier, projectCode, secondaryName,
413           desc, project.getSpaceCode(), new BeanItemContainer<ExperimentBean>(ExperimentBean.class),
414           new ProgressBar(), new Date(), "", "", null, false, false, false, "");
415 
416       // TODO isn't this slow in this fashion? what about SELECT * and creating a map?
417       String pi = datahandler.getDatabaseManager().getInvestigatorForProject(projectIdentifier);
418 
419       if (pi.equals("")) {
420         newProjectBean.setPrincipalInvestigator("n/a");
421       } else {
422         newProjectBean.setPrincipalInvestigator(pi);
423       }
424 
425       projectContainer.addBean(newProjectBean);
426     }
427 
428     homeSpaceBean.setProjects(projectContainer);
429     if (homeSpaceBean.getProjects().size() > 0) {
430       this.setContainerDataSource(homeSpaceBean, caption);
431     }
432   }
