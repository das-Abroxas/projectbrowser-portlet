486     filterTable.setRowHeaderMode(RowHeaderMode.INDEX);

499   @Override
500   public void enter(ViewChangeEvent event) {
501     Map<String, String> map = getMap(event.getParameters());
502     if (map == null)
503       return;
504     try {
505       HierarchicalContainer datasetContainer = new HierarchicalContainer();
506       datasetContainer.addContainerProperty("Select", CheckBox.class, null);
507       datasetContainer.addContainerProperty("Project", String.class, null);
508       datasetContainer.addContainerProperty("Sample", String.class, null);
509       // datasetContainer.addContainerProperty("Sample Type", String.class, null);
510       datasetContainer.addContainerProperty("File Name", String.class, null);
511       datasetContainer.addContainerProperty("File Type", String.class, null);
512       datasetContainer.addContainerProperty("Dataset Type", String.class, null);
513       datasetContainer.addContainerProperty("Registration Date", Timestamp.class, null);
514       datasetContainer.addContainerProperty("Validated", Boolean.class, null);
515       datasetContainer.addContainerProperty("File Size", String.class, null);
516       datasetContainer.addContainerProperty("file_size_bytes", Long.class, null);
517       datasetContainer.addContainerProperty("dl_link", String.class, null);
518       datasetContainer.addContainerProperty("CODE", String.class, null);
519 
520       List<ch.systemsx.cisd.openbis.generic.shared.api.v1.dto.DataSet> retrievedDatasets = null;
521 
522       switch (map.get("type")) {
523         case "project":
524           String projectIdentifier = map.get("id");
525           retrievedDatasets = datahandler.getOpenBisClient()
526               .getDataSetsOfProjectByIdentifierWithSearchCriteria(projectIdentifier);
527           break;
528 
529         case "experiment":
530           String experimentIdentifier = map.get("id");
531           retrievedDatasets = datahandler.getOpenBisClient()
532               .getDataSetsOfExperimentByCodeWithSearchCriteria(experimentIdentifier);
533           break;
534 
535         case "sample":
536           String sampleIdentifier = map.get("id");
537           String sampleCode = sampleIdentifier.split("/")[2];
538           retrievedDatasets = datahandler.getOpenBisClient().getDataSetsOfSample(sampleCode);
539           break;
540 
541         default:
542           retrievedDatasets =
543               new ArrayList<ch.systemsx.cisd.openbis.generic.shared.api.v1.dto.DataSet>();
544           break;
545       }
546 
547       numberOfDatasets = retrievedDatasets.size();
548       if (numberOfDatasets == 0) {
549         new Notification("No datasets available.", "<br/>Please contact the project manager.",
550             Type.WARNING_MESSAGE, true).show(Page.getCurrent());
551       } else {
552 
553         Map<String, String> samples = new HashMap<String, String>();
554 
555         // project same for all datasets
556         String projectCode = retrievedDatasets.get(0).getExperimentIdentifier().split("/")[2];
557         for (ch.systemsx.cisd.openbis.generic.shared.api.v1.dto.DataSet dataset : retrievedDatasets) {
558           samples.put(dataset.getCode(), dataset.getSampleIdentifierOrNull().split("/")[2]);
559         }
560 
561         List<DatasetBean> dsBeans = datahandler.queryDatasetsForFolderStructure(retrievedDatasets);
562 
563         for (DatasetBean d : dsBeans) {
564           Date date = d.getRegistrationDate();
565           SimpleDateFormat sd = new SimpleDateFormat("yyyy-MM-dd hh:mm:ss");
566           String dateString = sd.format(date);
567           Timestamp ts = Timestamp.valueOf(dateString);
568           String sampleID = samples.get(d.getCode());
569 
570           registerDatasetInTable(d, datasetContainer, projectCode, sampleID, ts, null);
571         }
572       }
573 
574       this.setContainerDataSource(datasetContainer);
575 
576     } catch (Exception e) {
577       e.printStackTrace();
578       LOG.error(String.format("getting dataset failed for dataset %s", map.toString()),
579           e.getStackTrace());
580     }
581   }

585   public void registerDatasetInTable(DatasetBean d, HierarchicalContainer dataset_container,
586       String project, String sample, Timestamp ts, Object parent) {
587     if (d.hasChildren()) {
588 
589       Object new_ds = dataset_container.addItem();

593 
594       dataset_container.setChildrenAllowed(new_ds, true);

597 
598       dataset_container.getContainerProperty(new_ds, "Project").setValue(project);
599       dataset_container.getContainerProperty(new_ds, "Sample").setValue(sample);
600       // dataset_container.getContainerProperty(new_ds, "Sample Type").setValue(

692     /**
693      * updates entries (puts and removes) for selected table item and all its children. Means
694      * Checkbox is updated. And in case download button is clicked all checked items will be
695      * downloaded.
696      * 
697      * @param itemId Container id
698      * @param itemSelected checkbox value of the item
699      * @param entries all checked items
700      * @param fileName fileName of current item
701      */
702     private void valueChange(Object itemId, boolean itemSelected,
703         Map<String, SimpleEntry<String, Long>> entries, String fileName) {
704 
705       ((CheckBox) table.getItem(itemId).getItemProperty("Select").getValue())
706           .setValue(itemSelected);
707       fileName = Paths
708           .get(fileName, (String) table.getItem(itemId).getItemProperty("File Name").getValue())
709           .toString();
710 
711       // System.out.println(fileName);
712       if (table.hasChildren(itemId)) {
713         for (Object childId : table.getChildren(itemId)) {
714           valueChange(childId, itemSelected, entries, fileName);
715         }
716       } else if (itemSelected) {
717         String datasetCode = (String) table.getItem(itemId).getItemProperty("CODE").getValue();
718         Long datasetFileSize =
719             (Long) table.getItem(itemId).getItemProperty("file_size_bytes").getValue();
720         entries.put(fileName,
721             new SimpleEntry<String, Long>(datasetCode, datasetFileSize));
722       } else {
723         entries.remove(fileName);
724       }
725     }

728   /**
729    * The input should have the following form: type=openbis_type&id=openbis_id e.g.
730    * type=sample&id=/ABI_SYSBIO/QMARI117AV It is specifically designed to be used in the case of
731    * datasetView. In other cases there is no guarantee that it will work correctly. returns a map
732    * with two entries: "type": "openbistype" "id" : "openbisId"
733    * 
734    * @param parameters
735    * @return
736    */
737   public static Map<String, String> getMap(String parameters) {
738     if (parameters == null || parameters.equals(""))
739       return null;
740     String[] params = parameters.split("&");
741     // TODO check for length == 2 needed ?
742     // if (params == null || params.length != 2)
743     if (params == null || params.length > 3)
744       return null;
745     HashMap<String, String> map = new HashMap<String, String>();
746     for (String p : params) {
747       String[] kv = p.split("=");
748       if (kv.length != 2)
749         return null;
750       map.put(kv[0], kv[1]);
751     }
752     return map;
753   }
