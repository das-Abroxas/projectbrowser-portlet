282   public List<Sample> querySamples(String queryString) {
283     EnumSet<SampleFetchOption> fetchOptions = EnumSet.of(SampleFetchOption.PROPERTIES,
284             SampleFetchOption.BASIC);
285 
286             //, SampleFetchOption.CONTAINED, SampleFetchOption.DESCENDANTS,
287             //SampleFetchOption.PARENTS, SampleFetchOption.CHILDREN, SampleFetchOption.ANCESTORS);
288 
289     SearchCriteria sc = new SearchCriteria();
290     sc.addMatchClause(MatchClause.createAnyFieldMatch("*" + queryString + "*"));
291 
292     List<Sample> samples = datahandler.getOpenBisClient().getOpenbisInfoService().searchForSamplesOnBehalfOfUser(datahandler.getOpenBisClient().getSessionToken(), sc, fetchOptions, PortalUtils.getNonNullScreenName());
293 
294     return samples;
295   }

298   public List<Experiment> queryExperiments(String queryString) {
299     EnumSet<ExperimentFetchOption> fetchOptions =
300         EnumSet.of(ExperimentFetchOption.PROPERTIES, ExperimentFetchOption.BASIC);
301 
302     SearchCriteria sc = new SearchCriteria();
303     sc.addMatchClause(MatchClause.createAnyFieldMatch("*" + queryString + "*"));
304 
305     List<Experiment> exps = datahandler.getOpenBisClient().getOpenbisInfoService()
306         .searchForExperiments(datahandler.getOpenBisClient().getSessionToken(), sc);
307 
308     List<Experiment> expFilteredByUser = new ArrayList<Experiment>();
309 
310     for (Experiment e : exps) {
311       String[] splitstr = e.getIdentifier().split("/");
312       String spaceCode = splitstr[1];
313 
314       if (visibleSpaces.contains(spaceCode)) {
315         expFilteredByUser.add(e);
316       }
317     }
318 
319     return expFilteredByUser;
320   }

322   public List<Project> queryProjects(String queryString) {
323     List<Project> result = new ArrayList<Project>();
324 
325     List<Project> projects =
326         new ArrayList<Project>(datahandler.getOpenBisClient().getOpenbisInfoService()
327             .listProjectsOnBehalfOfUser(datahandler.getOpenBisClient().getSessionToken(),
328                 PortalUtils.getNonNullScreenName()));
329 
330     for (Project p : projects) {
331       String projectDesc = p.getDescription();
332 
333       // sometimes there is no description available (check for null entry)
334       if (projectDesc != null && projectDesc.contains(queryString.toLowerCase())) {
335         result.add(p);
336       } else if (p.getCode().contains(queryString)) {
337         result.add(p);
338       }
339     }
340     return result;
341   }

344   public Set<String> listSpacesOnBehalfOfUser() {
345     // first retrieve all projects belonging to the user
346     List<Project> projects =
347         new ArrayList<Project>(datahandler.getOpenBisClient().getOpenbisInfoService()
348             .listProjectsOnBehalfOfUser(datahandler.getOpenBisClient().getSessionToken(),
349                 PortalUtils.getNonNullScreenName()));
350 
351     Set<String> resultSpaceNames = new HashSet<String>();
352 
353     for (Project p : projects) {
354       resultSpaceNames.add(p.getSpaceCode());
355       // LOG.info(p.getSpaceCode());
356     }
357 
358     // LOG.info(resultSpaceNames.toString());
359 
360     return resultSpaceNames;
361   }

364   public List<String> getSearchResults(String samplecode) {
365     EnumSet<SampleFetchOption> fetchOptions = EnumSet.of(SampleFetchOption.PROPERTIES);
366     SearchCriteria sc = new SearchCriteria();
367     sc.addMatchClause(
368         MatchClause.createAttributeMatch(MatchClauseAttribute.CODE, samplecode + "*"));
369 
370     List<Sample> samples = datahandler.getOpenBisClient().getOpenbisInfoService()
371         .searchForSamplesOnBehalfOfUser(datahandler.getOpenBisClient().getSessionToken(), sc,
372             fetchOptions, PortalUtils.getNonNullScreenName());
373     List<String> ret = new ArrayList<String>(samples.size());
374     for (Sample sample : samples) {
375       ret.add(sample.getCode());
376     }
377     return ret;
