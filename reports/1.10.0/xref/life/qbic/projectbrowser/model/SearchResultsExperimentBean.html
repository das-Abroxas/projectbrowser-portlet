52    public SearchResultsExperimentBean(Experiment e, String query) {
53      experimentID = e.getIdentifier();
54      experimentName = this.extractSampleProperty(e, "Q_SECONDARY_NAME");
55      queryString = query;
56      matchedField = findMatchedFields(e);

59    private String findMatchedFields(Experiment e) {
60  
61      StringBuilder strbuild = new StringBuilder();
62      List<String> queryComponents = Arrays.asList(queryString.trim().split("\\s+"));
63  
64      Map<String, String> props = e.getProperties();
65  
66      String code = e.getCode();
67  
68      for (String str : queryComponents) {
69        Pattern tmpPattern = Pattern.compile(str, Pattern.CASE_INSENSITIVE + Pattern.LITERAL);
70        Matcher patternMatch = tmpPattern.matcher(code);
71  
72        if (patternMatch.find()) {
73          strbuild.append("code: " + code + ",");
74        }
75  
76        for (String k : props.keySet()) {
77          if (k.equals("Q_SECONDARY_NAME")) {
78            continue;
79          }
80  
81          if (k.equals("Q_PROPERTIES")) {
82            Map<String, String> xmlPropsMap = new HashMap<String, String>();
83            try {
84              xmlPropsMap = parseXMLString(props.get(k).trim());
85            } catch (Exception ex) {
86              // TODO Auto-generated catch block
87              ex.printStackTrace();
88            }
89  
90            for (String k2 : xmlPropsMap.keySet()) {
91              patternMatch = tmpPattern.matcher(xmlPropsMap.get(k2).trim());
92  
93              if (patternMatch.find()) {
94                if (!"".equals(strbuild.toString())) {
95                  strbuild.append(", ");
96                }
97                strbuild.append(k2 + ": " + xmlPropsMap.get(k2));
98              }
99            }
100         } else {
101           patternMatch = tmpPattern.matcher(props.get(k));
102 
103           if (patternMatch.find()) {
104             if (!"".equals(strbuild.toString())) {
105               strbuild.append(", ");
106             }
107 
108             strbuild.append(k + ": " + props.get(k));
109           }
110         }
111       }
112     }
113 
114     String result = strbuild.toString();
115 
116     return result;
117   }

119   private Map<String, String> parseXMLString(String input) throws Exception {
120 
121     Map<String, String> xmlProps = new HashMap<String, String>();
122     Document xmlDoc = null;
123     try {
124       DocumentBuilderFactory factory = DocumentBuilderFactory.newInstance();
125       DocumentBuilder builder = factory.newDocumentBuilder();
126       InputSource is = new InputSource(new StringReader(input));
127       xmlDoc = builder.parse(is);
128 
129     } catch (Exception e) {
130       // TODO Auto-generated catch block
131       e.printStackTrace();
132     }
133 
134     if (xmlDoc != null) {
135       NodeList nodes = xmlDoc.getElementsByTagName("qcategorical");
136 
137       for (int i = 0; i < nodes.getLength(); ++i) {
138         Element element = (Element) nodes.item(i);
139 
140         String factorName = element.getAttribute("label");
141         String factorValue = element.getAttribute("value");
142         // System.out.println("Name: " + factorName + " " + factorValue);
143         xmlProps.put(factorName, factorValue);
144       }
145 
146     }
147 
148     return xmlProps;
149   }
