99     * Returns a Container with the informations of de.uni_tuebingen.qbic.beans.DatasetBean.
100    * 
101    * @param datasets
102    * @return
103    */
104   public Container fillTable(
105       List<ch.systemsx.cisd.openbis.generic.shared.api.v1.dto.DataSet> datasets, String projectID) {
106     HashMap<String, ch.systemsx.cisd.openbis.generic.shared.api.v1.dto.DataSet> dataMap =
107         new HashMap<String, ch.systemsx.cisd.openbis.generic.shared.api.v1.dto.DataSet>();
108     BeanItemContainer<DatasetBean> container =
109         new BeanItemContainer<DatasetBean>(DatasetBean.class);
110 
111     for (ch.systemsx.cisd.openbis.generic.shared.api.v1.dto.DataSet ds : datasets) {
112       dataMap.put(ds.getCode(), ds);
113     }
114 
115     List<life.qbic.projectbrowser.model.DatasetBean> datasetBeans =
116         new ArrayList<life.qbic.projectbrowser.model.DatasetBean>();
117     datasetBeans = datahandler.queryDatasetsForFiles(datasets);
118 
119     List<String> fileNames = new ArrayList<String>();
120 
121     for (life.qbic.projectbrowser.model.DatasetBean bean : datasetBeans) {
122       fileNames.add(bean.getFileName());
123 
124       DatasetBean newBean = new DatasetBean(bean.getFileName(),
125           dataMap.get(bean.getCode()).getDataSetTypeCode(), bean.getCode(), bean.getDssPath(),
126           dataMap.get(bean.getCode()).getSampleIdentifierOrNull());
127 
128       if (dataMap.get(bean.getCode()).getProperties() != null) {
129         newBean.setProperties(dataMap.get(bean.getCode()).getProperties());
130       }
131 
132       container.addBean(newBean);
133     }
134 
135     this.projectID = projectID;
136     this.fileNames = fileNames;
137 
138     return container;
139   }

141   /**
142    * Register a new Workflow experiment in openBIS. Should be done when starting the workflow.
143    * Experiment name is automatically created from the project name and the number of existing
144    * experiments in that project. Standard workflow experiment fields are also initialized.
145    * 
146    * @param space space code
147    * @param project project code
148    * @param typecode openbis type code of the workflow
149    * @param wfName name of the workflow
150    * @param wfVersion version of the workflow
151    * @param userID the user that starts the workflow
152    * @param qProperties
153    * @return Code of the newly registered experiment
154    */
155   public String registerWFExperiment(String space, String project, String typecode, String wfName,
156       String wfVersion, String userID, String qProperties) {
157     int last = 0;
158     for (Experiment e : datahandler.getOpenBisClient().getExperimentsOfProjectByIdentifier(
159         (new StringBuilder("/")).append(space).append("/").append(project).toString())) {
160       String[] codeSplit = e.getCode().split("E");
161       String number = codeSplit[codeSplit.length - 1];
162       int num = 0;
163       try {
164         num = Integer.parseInt(number);
165       } catch (NumberFormatException ex) {
166       }
167       last = Math.max(num, last);
168     }
169 
170     LOG.debug("Space: " + space);
171     LOG.debug("Project: " + project);
172     LOG.debug("StringBuilder: "
173         + new StringBuilder("/").append(space).append("/").append(project).toString());
174 
175     String code = project + "E" + Integer.toString(last + 1);
176 
177     LOG.debug("Code: " + code);
178 
179     Map<String, Object> params = new HashMap<String, Object>();
180     params.put("code", code);
181     params.put("type", typecode);
182     params.put("project", project);
183     params.put("space", space);
184 
185     Map<String, Object> properties = new HashMap<String, Object>();
186     properties.put(wf_name, wfName);
187     properties.put(wf_version, wfVersion);
188     properties.put(wf_executer, userID);
189     properties.put(wf_started, Utils.getTime());
190     properties.put(wf_status, workflow_statuses.RUNNING.toString());
191     properties.put("Q_PROPERTIES", qProperties);
192 
193     params.put("properties", properties);
194 
195     datahandler.getOpenBisClient().ingest(openbis_dss, "register-exp", params);
196     return code;
197   }

208   public String registerWFSample(String space, String project, String experiment, String typecode,
209       List<String> parents, List<DatasetBean> datasets) {
210     int last = 0;
211     for (Sample s : datahandler.getOpenBisClient().getSamplesofExperiment((new StringBuilder("/"))
212         .append(space).append("/").append(project).append("/").append(experiment).toString())) {
213       String[] codeSplit = s.getCode().split("R");
214       String number = codeSplit[codeSplit.length - 1];
215       int num = 0;
216       try {
217         num = Integer.parseInt(number);
218       } catch (NumberFormatException ex) {
219       }
220       last = Math.max(num, last);
221     }
222 
223     String code =
224         (new StringBuilder(experiment)).append("R").append(Integer.toString(last + 1)).toString();
225     Map<String, Object> params = new HashMap<String, Object>();
226     params.put("code", code);
227     params.put("type", typecode);
228 
229     params.put("sample_class", "");
230     params.put("parents", parents);
231 
232     params.put("project", project);
233     params.put("space", space);
234     params.put("experiment", experiment);
235 
236     StringBuilder result = new StringBuilder();
237     result.append("Input Files: ");
238 
239     for (DatasetBean b : datasets) {
240       result.append(b.getFileName());
241       result.append(",");
242     }
243 
244     String secName = result.length() > 0 ? result.substring(0, result.length() - 1) : "";
245 
246     Map<String, Object> properties = new HashMap<String, Object>();
247     properties.put("Q_ADDITIONAL_INFO", secName);
248     params.put("properties", properties);
249 
250     datahandler.getOpenBisClient().ingest(openbis_dss, "register-samp", params);
251     return code;
252   }

351   private void mapExperimentalProperties(String id, List<String> fileNames) {
352     Map<String, Object> params = new HashMap<String, Object>();
353     List<String> codes = new ArrayList<String>();
354     for (Sample s : datahandler.getOpenBisClient().getSamplesOfProjectBySearchService(id))
355       codes.add(s.getCode());
356     params.put("codes", codes);
357     QueryTableModel res =
358         datahandler.getOpenBisClient().getAggregationService("get-property-tsv", params);
359 
360     Set<String> factorNames = new HashSet<String>();
361     Map<String, String> fileProps = new HashMap<String, String>();
362 
363     Set<String> secondaryNames = new HashSet<String>();
364     Set<String> factorLabels = datahandler.getFactorLabels();
365 
366     for (Serializable[] ss : res.getRows()) {
367 
368       String sampleCode = (String) ss[0];
369       List<Property> properties = new ArrayList<Property>();
370       for (String label : factorLabels) {
371         Property f =
372             datahandler.getFactorsForLabelsAndSamples().get(new ImmutablePair<>(label, sampleCode));
373         if (f != null) {
374           properties.add(f);
375         }
376       }
377 
378       List<String> matches = getMatchingStrings(fileNames, sampleCode);
379       if (!matches.isEmpty()) {
380         for (String match : matches) {
381           StringBuilder row = new StringBuilder();
382           String extID = (String) ss[1];// how to use this if it is preferred over secondary name?
383           String secondaryName = (String) ss[2];
384           while (secondaryNames.contains(secondaryName))
385             secondaryName += "1";
386           secondaryNames.add(secondaryName);
387           row.append(secondaryName);
388           for (Property f : properties) {
389             factorNames.add(f.getLabel());
390             String val = f.getValue();
391             if (f.hasUnit())
392               val += f.getUnit();
393             row.append("\t" + val);
394           }
395           fileProps.put(match, row.toString());
396         }
397       }
398     }
399     this.expProps = fileProps;
400     this.expFactors = factorNames;
401   }

403   /**
404    * Finds the the matching strings in the list, e.g. sample codes in file names
405    * 
406    * @param list The list of strings to check
407    * @param substring The regular expression to use
408    * @return List of matching Strings
409    */
410   static List<String> getMatchingStrings(List<String> list, String substring) {
411     List<String> res = new ArrayList<String>();
412     for (String s : list) {
413       if (s.contains(substring)) {
414         res.add(s);
415       }
416     }
417     return res;
418   }

424   public String submitAndRegisterWf(String type, String id, Workflow workflow,
425       List<DatasetBean> selectedDatasets)
426       throws ConnectException, IllegalArgumentException, SubmitFailedException {
427 
428     SpaceAndProjectCodes spaceandproject = getSpaceAndProjects(type, id);
429 
430     String spaceCode = spaceandproject.space;
431     String projectCode = spaceandproject.project;
432 
433     ParameterSet params = workflow.getParameters();
434     List<Property> factors = new ArrayList<Property>();
435 
436     XMLParser xmlParser = new XMLParser();
437 
438     for (Map.Entry<String, Parameter> entry : workflow.getData().getData().entrySet()) {
439       String key = entry.getKey();
440       Parameter value = entry.getValue();
441 
442       if (key.contains("input")) {
443         List<String> files = (List<String>) value.getValue();
444         List<String> inputFiles = new ArrayList<String>();
445 
446         for (String f : files) {
447           String[] splitted = f.split("/");
448           String fileName = splitted[splitted.length - 1];
449           inputFiles.add(fileName);
450         }
451 
452         System.out.println(inputFiles.toString());
453         String concat = String.join("; ", inputFiles);
454         System.out.println(concat);
455         Property newProperty = new Property("input_files", concat, PropertyType.Property);
456         factors.add(newProperty);
457       }
458 
459       else {
460         Property newProperty = new Property("database",
461             value.getValue().toString().replace("/lustre_cfc/qbic/reference_genomes/", ""),
462             PropertyType.Property);
463         factors.add(newProperty);
464       }
465     }
466 
467     for (String p : params.getParamNames()) {
468       Parameter par = params.getParam(p);
469       String[] splitted = par.getTitle().split("\\.");
470       String parName = splitted[splitted.length - 1].replace(" ", "_").toLowerCase();
471 
472       Property newProperty =
473           new Property(parName, par.getValue().toString(), PropertyType.Property);
474       factors.add(newProperty);
475     }
476 
477     String qProperties = "";
478 
479     try {
480       qProperties = xmlParser.toString(xmlParser.createXMLFromProperties(factors));
481       System.out.println(qProperties);
482     } catch (JAXBException e) {
483       // TODO Auto-generated catch block
484       e.printStackTrace();
485     }
486 
487     String experimentCode = registerWFExperiment(spaceCode, projectCode,
488         workflow.getExperimentType(), workflow.getID(), workflow.getVersion(), user, qProperties);
489 
490     List<String> parents = getConnectedSamples(selectedDatasets);
491     String sampleType = workflow.getSampleType();
492 
493     String sampleCode = registerWFSample(spaceCode, projectCode, experimentCode, sampleType,
494         parents, selectedDatasets);
495 
496     String openbisId =
497         String.format("%s-%s-%s-%s", spaceCode, projectCode, experimentCode, sampleCode);
498 
499     LOG.info("User: " + user + " is submitting workflow " + workflow.getID() + " openbis id is:"
500         + openbisId);
501 
502     String submit_id = submitter.submit(workflow, openbisId, user);
503     LOG.info("Workflow has guse id: " + submit_id);
504 
505     setWorkflowID(spaceCode, projectCode, experimentCode, submit_id);
506     return openbisId;
507   }

509   private SpaceAndProjectCodes getSpaceAndProjects(String type, String id) {
510     String[] split = id.split("/");
511 
512     if (split.length == 0)
513       return null;
514     switch (type) {
515       case PatientView.navigateToLabel:
516       case ProjectView.navigateToLabel:
517       case ExperimentView.navigateToLabel:
518       case "workflowExperimentType":
519         return new SpaceAndProjectCodes(split[1], split[2]);
520       case SampleView.navigateToLabel:
521         String expId = datahandler.getOpenBisClient()
522             .getSampleByIdentifier(String.format("%s/%s", split[1], split[2]))
523             .getExperimentIdentifierOrNull();
524         if (expId == null)
525           return null;
526         return getSpaceAndProjects(ExperimentView.navigateToLabel, expId);
527       case DatasetView.navigateToLabel:
528         throw new NotImplementedException("Dataset view is not ready for workflows!");
529       default:
530         LOG.debug(String.format("Problem with id %s, type %s", id, type));
531         return null;
532     }
533   }

546   /**
547    * get for a {@link de.uni_tuebingen.qbic.beans.DatasetBean} file or directory the path it has on
548    * the data store server.
549    * 
550    * @param bean
551    * @return full path of dataset
552    * @throws IllegalArgumentException
553    */
554   public String getDatasetsNfsPath(DatasetBean bean) throws IllegalArgumentException {
555 
556     try {
557       DataSet dataset =
558           datahandler.getOpenBisClient().getFacade().getDataSet(bean.getOpenbisCode());
559       String path = dataset.getDataSetDss().tryGetInternalPathInDataStore();
560 
561       if (bean.getFullPath().startsWith("original")) {
562         path = Paths.get(path, bean.getFullPath()).toString();
563       } else {
564         FileInfoDssDTO[] filelist = dataset.listFiles("original", false);
565         path = path + "/original/" + filelist[0].getPathInListing();
566       }
567       // TODO get rid of hardocded paths
568       path = path.replaceFirst("/mnt/" + openbis_dss, "/mnt/nfs/qbic");
569       path = path.replaceFirst("/mnt/DSS_icgc", "/mnt/glusterfs/DSS_icgc");
570       path = path.replaceFirst("/beegfs/bigbio", "/mnt/nfs/bigbio_temp");
571       return path;
572     } catch (Exception e) {
573       e.printStackTrace();
574       throw new IllegalArgumentException("Could not retrieve nfs path for dataset " + bean);
575     }
576   }
