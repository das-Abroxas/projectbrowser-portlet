16  package life.qbic.projectbrowser.helpers;
17  
18  import java.io.ByteArrayInputStream;
19  import java.io.InputStream;
20  import java.util.ArrayList;
21  import java.util.List;
22  import java.util.Map;
23  
24  import com.vaadin.server.StreamResource;
25  
26  
27  import life.qbic.projectbrowser.components.TSVDownloadComponent;
28  
29  public class TSVReadyRunnable implements Runnable {
30  
31    private TSVDownloadComponent layout;
32    Map<String, String> tableContentStrings;
33    private String project;
34  
35    public TSVReadyRunnable(TSVDownloadComponent layout, Map<String, String> tableContentStrings,
36        String project) {
37      this.layout = layout;
38      this.tableContentStrings = tableContentStrings;
39      this.project = project;
40    }
41  
42    @Override
43    public void run() {
44      List<StreamResource> streams = new ArrayList<StreamResource>();
45      streams.add(
46          getTSVStream(tableContentStrings.get("Q_BIOLOGICAL_ENTITY"), project + "_sample_sources"));
47      streams.add(
48          getTSVStream(tableContentStrings.get("Q_BIOLOGICAL_SAMPLE"), project + "_sample_extracts"));
49      if (tableContentStrings.containsKey("Q_TEST_SAMPLE"))
50        streams.add(
51            getTSVStream(tableContentStrings.get("Q_TEST_SAMPLE"), project + "_sample_preparations"));
52      layout.armButtons(streams);
53    }
54  
55    public StreamResource getTSVStream(final String content, String name) {
56      StreamResource resource = new StreamResource(new StreamResource.StreamSource() {
57        @Override
58        public InputStream getStream() {
59          try {
60            InputStream is = new ByteArrayInputStream(content.getBytes());
61            return is;
62          } catch (Exception e) {
63            e.printStackTrace();
64            return null;
65          }
66        }
67      }, String.format("%s.tsv", name));
68      return resource;
69    }
70  }
