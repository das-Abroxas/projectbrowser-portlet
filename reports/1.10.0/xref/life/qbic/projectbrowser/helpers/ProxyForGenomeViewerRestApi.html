50    private static final long serialVersionUID = 3929423697741337860L;
51    private long session;
52  
53    public ProxyForGenomeViewerRestApi() {
54      Random rand = new Random();
55      rand.setSeed(
56          System.nanoTime() + Long.parseLong(PortalUtils.getNonNullScreenName(), 36));
57      session = rand.nextLong();
58      UI.getCurrent().getSession().setAttribute("gv-restapi-session", session);
59    }
60  
61    @Override
62    public boolean handleRequest(VaadinSession session, VaadinRequest request,
63        VaadinResponse response) throws IOException {
64      if (UI.getCurrent() != null && UI.getCurrent().getPage() != null
65          && UI.getCurrent().getPage().getUriFragment() != null) {
66        System.out.println(UI.getCurrent().getPage().getUriFragment());
67      }
68      System.out.println("is handling request");
69      System.out.println(request.getPathInfo());
70      System.out.println(request.getContextPath());
71      PortletRequest portletRequest = VaadinPortletService.getCurrentPortletRequest();
72      Map<String, String[]> para = request.getParameterMap();
73      Set<Entry<String, String[]>> s = para.entrySet();
74      Iterator<Entry<String, String[]>> it = s.iterator();
75      while (it.hasNext()) {
76        Entry<String, String[]> en = it.next();
77        System.out.println("Key: " + en.getKey());
78        System.out.println("Value:");
79        for (int i = 0; i < en.getValue().length; i++) {
80          System.out.println(en.getValue()[i]);
81        }
82        System.out.println("Next");
83      }
84      Enumeration<String> enu = portletRequest.getParameterNames();
85      while (enu.hasMoreElements()) {
86        String su = enu.nextElement();
87        System.out.println(su + " " + portletRequest.getParameter(su));
88      }
89      enu = portletRequest.getPropertyNames();
90      while (enu.hasMoreElements()) {
91        String su = enu.nextElement();
92        System.out.println(su + " " + portletRequest.getProperty(su));
93      }
94      HttpServletRequest httpRequest = PortalUtil.getHttpServletRequest(portletRequest);
95      System.out.println(httpRequest.getPathInfo());
96      enu = httpRequest.getParameterNames();
97      while (enu.hasMoreElements()) {
98        String su = enu.nextElement();
99        System.out.println(su + " " + httpRequest.getParameter(su));
100     }
101     System.out.println(httpRequest.getQueryString());
102     enu = PortalUtil.getOriginalServletRequest(httpRequest).getParameterNames();
103     System.out.println(PortalUtil.getOriginalServletRequest(httpRequest).getPathInfo());
104     while (enu.hasMoreElements()) {
105       String su = enu.nextElement();
106       System.out
107           .println(su + " " + PortalUtil.getOriginalServletRequest(httpRequest).getParameter(su));
108     }
109 
110 
111     if (!String.valueOf(session).equals(request.getPathInfo())) {
112       return false;
113     }
114 
115     String fileId = request.getParameter("fileId");
116     String filepaths = request.getParameter("filepath");
117     String removeZeroGenotypes = request.getParameter("removeZeroGenotypes");
118     String region = request.getParameter("region");
119     String interval = request.getParameter("interval");
120     String histogram = request.getParameter("histogram");
121 
122     URL u;
123     try {
124       StringBuilder sb = new StringBuilder();
125       sb.append(ConfigurationManagerFactory.getInstance().getGenomeViewerRestApiUrl());
126       sb.append(fileId);
127       sb.append("/fetch?filepaths=");
128       sb.append(filepaths);
129       sb.append("&region=");
130       sb.append(region);
131       if (interval != null) {
132         sb.append("&interval=");
133         sb.append(interval);
134       }
135       if (histogram != null) {
136         sb.append("&histogram=");
137         sb.append(histogram);
138       }
139       if (removeZeroGenotypes != null) {
140         sb.append("");
141       }
142       u = new URL(sb.toString());
143       // u = new
144       // URL("http://localhost:7777/vizrest/rest/data/QBAMS001AB.bam/fetch?filepaths=/store/1/0EEF79A2-8140-4FC7-BA67-E51908FE4619/f0/91/36/20141116104428925-3161/original/QBAMS001AB.bam&removeZeroGenotypes=false&region=9:117163200-117164799,9:117164800-117166399,9:117166400-117167999,9:117168000-117169599&interval=1600&h");
145       URLConnection urlConnection = u.openConnection();
146 
147       urlConnection.setUseCaches(false);
148       urlConnection.setDoOutput(false);
149       urlConnection.connect();
150       InputStream is = urlConnection.getInputStream();
151       response.setContentType("application/json");
152       response.setHeader("Content-Type", "application/json");
153       OutputStream out = response.getOutputStream();
154       byte[] buffer = new byte[com.vaadin.server.Constants.DEFAULT_BUFFER_SIZE];
155       while (true) {
156         int readCount = is.read(buffer);
157         if (readCount < 0) {
158           break;
159         }
160         out.write(buffer, 0, readCount);
161       }
162 
163     } catch (MalformedURLException e) {
164       // TODO Auto-generated catch block
165       e.printStackTrace();
166     } catch (IOException e) {
167       // TODO Auto-generated catch block
168       e.printStackTrace();
169     }
170     return true;
171   }
