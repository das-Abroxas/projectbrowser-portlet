21  import java.util.List;

23  /**
24   * Helper functions used for sample creation
25   * 
26   * @author Andreas Friedrich
27   * 
28   */
29  public class BarcodeFunctions {
30  
31    public static float getPercentageStep(int max) {
32      return new Float(1.0 / max);
33    }
34  
35    /**
36     * Computes a checksum digit for a given String. This checksum is weighted position-specific,
37     * meaning it will also most likely fail a check if there is a typo of the String resulting in a
38     * swapping of two numbers.
39     * 
40     * @param s String for which a checksum should be computed.
41     * @return Character representing the checksum of the input String.
42     */
43    public static char checksum(String s) {
44      int i = 1;
45      int sum = 0;
46      for (int idx = 0; idx <= s.length() - 1; idx++) {
47        sum += (((int) s.charAt(idx))) * i;
48        i += 1;
49      }
50      return Utils.mapToChar(sum % 34);
51    }
52  
53    /**
54     * Returns a String denoting the range of a list of barcodes as used in QBiC
55     * 
56     * @param ids List of code strings
57     * @return String denoting a range of the barcodes
58     */
59    public static String getBarcodeRange(List<String> ids) {
60      String head = OpenBisFunctions.getProjectPrefix(ids.get(0));
61      String min = ids.get(0).substring(5, 8);
62      String max = min;
63      for (String id : ids) {
64        String num = id.substring(5, 8);
65        if (num.compareTo(min) < 0)
66          min = num;
67        if (num.compareTo(max) > 0)
68          max = num;
69      }
70      return head + min + "-" + max;
71    }
72  
73    /**
74     * Checks if a String fits the QBiC barcode pattern
75     * 
76     * @param code A String that may be a barcode
77     * @return true if String is a QBiC barcode, false if not
78     */
79    public static boolean isQbicBarcode(String code) {
80      String pattern = "Q[A-Z0-9]{4}[0-9]{3}[A-Z0-9]{2}";
81      return code.matches(pattern);
82    }
83    
84    public static boolean isExperimentCode(String code) {
85      String pattern = "Q[A-Z0-9]{4}E[1-9][0-9]*";
86      return code.matches(pattern);
87    }
88  }
