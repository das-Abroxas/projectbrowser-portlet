18  package life.qbic.projectbrowser.model;
19  import java.util.Map;
20  
21  import com.vaadin.ui.CustomComponent;
22  import com.vaadin.ui.HorizontalLayout;
23  import com.vaadin.ui.Label;
24  
25  import ch.systemsx.cisd.openbis.generic.shared.api.v1.dto.Sample;
26  
27  public class SearchResultsSampleItem extends CustomComponent {
28  	Sample sampleToView;
29  
30  	public SearchResultsSampleItem(Sample sampleToView, int rowNumber) {
31  		this.sampleToView = sampleToView;
32  
33  		Map<String, String> props = sampleToView.getProperties();
34  		
35  		HorizontalLayout rowContent = new HorizontalLayout();
36  		rowContent.setSpacing(true);
37  		Label itemNumber = new Label(Integer.toString(rowNumber) + ".");
38  		Label spacer1 = new Label();
39  		Label qbicCode = new Label(sampleToView.getCode());
40  		Label spacer2 = new Label();
41  		
42  		String sampleLabel;
43  		
44  		if (props.get("Q_SECONDARY_NAME") == null || props.get("Q_SECONDARY_NAME").equals("")) {
45  			sampleLabel = "no description";			
46  		}
47  		else {
48  			sampleLabel = props.get("Q_SECONDARY_NAME");
49  		}
50  		
51  		Label qbicLabel = new Label(sampleLabel);
52  		
53  		rowContent.addComponent(itemNumber);
54  
55  		rowContent.addComponent(qbicCode);
56  
57  		rowContent.addComponent(qbicLabel);
58  
59  
60  
61  		setCompositionRoot(rowContent);
62  
63  	}
64  
65  
66  }
