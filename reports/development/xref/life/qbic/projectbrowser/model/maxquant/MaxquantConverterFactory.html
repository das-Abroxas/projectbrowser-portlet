18  package life.qbic.projectbrowser.model.maxquant;
19  
20  import com.vaadin.data.util.converter.Converter;
21  import com.vaadin.data.util.converter.DefaultConverterFactory;
22  
23  /**
24   * According to this https://vaadin.com/forum#!/thread/8216422 The Converter (in this case
25   * MaxquantDigestionModeToNameConverter) will not call for Native select by defaut So one has to
26   * create a Factory and include it into VaadinSession via
27   * VaadinSession.getCurrent().setConverterFactory( new MaxquantConverterFactory()); this is true for
28   * vaadin 7.5
29   * 
30   * @author wojnar
31   * 
32   */
33  public class MaxquantConverterFactory extends DefaultConverterFactory {
34    private static final long serialVersionUID = 6988206501368446880L;
35  
36    @SuppressWarnings("unchecked")
37    @Override
38    public <PRESENTATION, MODEL> Converter<PRESENTATION, MODEL> createConverter(
39        Class<PRESENTATION> presentationType, Class<MODEL> modelType) {
40      // Handle one particular type conversion
41      if ((String.class == presentationType || Object.class == presentationType)
42          && DigestionMode.class == modelType)
43        return (Converter<PRESENTATION, MODEL>) new MaxquantDigestionModeToNameConverter();
44  
45      // Default to the supertype
46      return super.createConverter(presentationType, modelType);
47    }
48  }
