102   public List<Person> getPersonWithAffiliations(Integer personID) {
103     List<Person> res = new ArrayList<Person>();
104     String lnk = "persons_organizations";
105     String sql =
106         "SELECT persons.*, organizations.*, " + lnk + ".occupation FROM persons, organizations, "
107             + lnk + " WHERE persons.id = " + Integer.toString(personID) + " AND persons.id = "
108             + lnk + ".person_id and organizations.id = " + lnk + ".organization_id";
109     Connection conn = login();
110     try (PreparedStatement statement = conn.prepareStatement(sql)) {
111       ResultSet rs = statement.executeQuery();
112       while (rs.next()) {
113         int id = rs.getInt("id");
114         String username = rs.getString("username");
115         String title = rs.getString("title");
116         String first = rs.getString("first_name");
117         String last = rs.getString("family_name");
118         String eMail = rs.getString("email");
119         String phone = rs.getString("phone");
120 
121 
122         int affiliationID = rs.getInt("organizations.id");
123 
124         String group_acronym = rs.getString("group_acronym");
125         String group_name = rs.getString("group_name");
126         String institute = rs.getString("institute");
127         String organization = rs.getString("umbrella_organization");
128         String affiliation = "";
129 
130         if (group_name == null || group_name.toUpperCase().equals("NULL") || group_name.equals("")) {
131 
132           if (institute == null || institute.toUpperCase().equals("NULL") || institute.equals("")) {
133             affiliation = organization;
134           } else {
135             affiliation = institute;
136           }
137 
138         } else {
139           affiliation = group_name;
140           if (group_acronym != null && !group_acronym.isEmpty())
141             affiliation += " (" + group_acronym + ")";
142         }
143 
144         String role = rs.getString(lnk + ".occupation");
145         res.add(new Person(id, username, title, first, last, eMail, phone, affiliationID,
146             affiliation, role));
147       }
148       statement.close();
149     } catch (SQLException e) {
150       e.printStackTrace();
151     }
152     logout(conn);
153     return res;
154   }

156   public String getPersonDetailsForProject(String projectIdentifier, String role) {
157     String sql =
158         "SELECT projects_persons.*, projects.* FROM projects_persons, projects WHERE projects.openbis_project_identifier = ?"
159             + " AND projects.id = projects_persons.project_id AND projects_persons.project_role = ?";
160 
161     int id = -1;
162 
163     List<Person> personWithAffiliations = new ArrayList<Person>();
164 
165     Connection conn = login();
166     try (PreparedStatement statement = conn.prepareStatement(sql)) {
167       statement.setString(1, projectIdentifier);
168       statement.setString(2, role);
169 
170       ResultSet rs = statement.executeQuery();
171 
172       while (rs.next()) {
173         id = rs.getInt("person_id");
174       }
175       personWithAffiliations = getPersonWithAffiliations(id);
176     } catch (SQLException e) {
177       e.printStackTrace();
178       logout(conn);
179       // LOG.debug("Project not associated with Investigator. PI will be set to 'Unknown'");
180     }
181 
182     String details = "";
183     if (personWithAffiliations.size() > 0) {
184       Person p = personWithAffiliations.get(0);
185       String institute = p.getOneAffiliationWithRole().getAffiliation();
186 
187       String title = "";
188 
189       if (p.getTitle() != null) {
190         title = p.getTitle();
191       }
192 
193       details =
194           String.format("<p>%s %s %s <br> %s <br><br> %s <br> %s</p>", title, p.getFirst(),
195               p.getLast(), institute, p.getPhone(), p.geteMail());
196       // TODO is address important?
197     }
198 
199     logout(conn);
200     return details;
201   }

203   public boolean changeLongProjectDescription(String projectIdentifier, String description) {
204     LOG.info("Adding/Updating long description of project " + projectIdentifier);
205     boolean saved = saveOldDescription(projectIdentifier);
206     if(! saved)
207       LOG.warn("Could not save old project description to database!");
208     String sql = "UPDATE projects SET long_description = ? WHERE openbis_project_identifier = ?";
209     Connection conn = login();
210     PreparedStatement statement = null;
211     int res = -1;
212     try {
213       statement = conn.prepareStatement(sql);
214       statement.setString(1, description);
215       statement.setString(2, projectIdentifier);
216       statement.execute();
217       res = statement.getUpdateCount();
218       LOG.info("Successful.");
219     } catch (SQLException e) {
220       LOG.error("SQL operation unsuccessful: " + e.getMessage());
221       e.printStackTrace();
222     } finally {
223       endQuery(conn, statement);
224     }
225     return res != -1;
226   }

228   private boolean saveOldDescription(String projectIdentifier) {
229     String sql = "SELECT * from projects WHERE openbis_project_identifier = ?";
230     int id = -1;
231     String oldDescription = "";
232     String oldTitle = "";
233     Connection conn = login();
234     PreparedStatement statement = null;
235     try {
236       statement = conn.prepareStatement(sql);
237       statement.setString(1, projectIdentifier);
238       ResultSet rs = statement.executeQuery();
239       if (rs.next()) {
240         id = rs.getInt("id");
241         oldDescription = rs.getString("long_description");
242         oldTitle = rs.getString("short_title");
243       }
244     } catch (SQLException e) {
245       LOG.error("SQL operation unsuccessful: " + e.getMessage());
246       e.printStackTrace();
247     }
248     Date date = new Date();
249     Timestamp timestamp = new Timestamp(date.getTime());
250     sql = "INSERT INTO projects_history (project_id, timestamp, long_description, short_title) VALUES(?, ?, ?, ?)";
251     statement = null;
252     int res = -1;
253     try {
254       statement = conn.prepareStatement(sql);
255       statement.setInt(1, id);
256       statement.setTimestamp(2, timestamp);
257       statement.setString(3, oldDescription);
258       statement.setString(4, oldTitle);
259       statement.execute();
260       res = statement.getUpdateCount();
261       LOG.info("Successful.");
262     } catch (SQLException e) {
263       LOG.error("SQL operation unsuccessful: " + e.getMessage());
264       e.printStackTrace();
265     } finally {
266       endQuery(conn, statement);
267     }
268     return res != -1;
269   }

271   private void endQuery(Connection c, PreparedStatement p) {
272     if (p != null)
273       try {
274         p.close();
275       } catch (Exception e) {
276         LOG.error("PreparedStatement close problem");
277       }
278     if (c != null)
279       try {
280         logout(c);
281       } catch (Exception e) {
282         LOG.error("Database Connection close problem");
283       }
284   }
