107   public void updateUI(final String id, String type) {
108     propLayout.removeAllComponents();
109     Button saveButton = new Button("Submit Changes");
110     saveButton.setStyleName(ValoTheme.BUTTON_FRIENDLY);
111 
112     completeProperties = datahandler.getOpenBisClient()
113         .listPropertiesForType(datahandler.getOpenBisClient().getSampleTypeByString(type));
114 
115     sample = datahandler.getOpenBisClient().getSampleByIdentifier(id);
116     assignedProperties = sample.getProperties();
117 
118     saveButton.addClickListener(new ClickListener() {
119       @Override
120       public void buttonClick(final ClickEvent event) {
121         HashMap<String, Object> props = new HashMap<String, Object>();
122         Collection<Field<?>> registeredFields = fieldGroup.getFields();
123         // XMLParser xmlParser = new XMLParser();
124 
125         List<Property> factors = new ArrayList<Property>();
126 
127         boolean qpropertiesDefined = false;
128 
129         for (Field<?> field : registeredFields) {
130           if (field.getDescription().equals("Q_PROPERTIES")) {
131             TextField tf = (TextField) field;
132             qpropertiesDefined = true;
133             String label = tf.getCaption();
134             String val = (String) tf.getValue();
135             String[] splt = label.split(" in ");
136             Property f = null;
137             life.qbic.xml.properties.PropertyType type =
138                 (life.qbic.xml.properties.PropertyType) tf.getData();
139             if (splt.length > 1) {
140               label = splt[0];
141               life.qbic.xml.properties.Unit unit = life.qbic.xml.properties.Unit.valueOf(splt[1]);
142               f = new Property(label, val, unit, type);
143             } else {
144               f = new Property(label, val, type);
145             }
146             factors.add(f);
147           }
148 
149           else {
150             props.put(field.getDescription(), field.getValue());
151           }
152         }
153 
154         String user = PortalUtils.getNonNullScreenName();
155 
156         if (qpropertiesDefined) {
157           String qProperties = "";
158 
159 
160           JAXBElement<Qexperiment> setup = datahandler.getExperimentalSetup();
161           Map<String, Map<Pair<String, String>, List<String>>> factorStructure = new HashMap<>();
162           Map<String, List<Qproperty>> otherProps = new HashMap<>();
163           String code = sample.getCode();
164           for (Property p : factors) {
165             String lab = p.getLabel();
166             String val = p.getValue();
167             String unit = null;
168             if (p.hasUnit())
169               unit = p.getUnit().getValue();
170             if (p.getType().equals(life.qbic.xml.properties.PropertyType.Factor)) {
171 
172               Pair<String, String> valunit = new ImmutablePair<String, String>(val, unit);
173               if (factorStructure.containsKey(lab)) {
174                 Map<Pair<String, String>, List<String>> levels = factorStructure.get(lab);
175                 if (levels.containsKey(valunit)) {
176                   levels.get(valunit).add(code);
177                 } else {
178                   levels.put(valunit, new ArrayList<String>(Arrays.asList(code)));
179                 }
180               } else {
181                 Map<Pair<String, String>, List<String>> newLevel =
182                     new HashMap<Pair<String, String>, List<String>>();
183                 newLevel.put(valunit, new ArrayList<String>(Arrays.asList(code)));
184                 factorStructure.put(lab, newLevel);
185               }
186 
187             } else {
188               Qproperty newProp = null;
189               if (p.hasUnit()) {
190                 newProp = new Qproperty(code, p.getLabel(), p.getValue(), p.getUnit());
191               } else {
192                 newProp = new Qproperty(code, p.getLabel(), p.getValue());
193               }
194               if (otherProps.containsKey(code)) {
195                 otherProps.get(code).add(newProp);
196               } else {
197                 otherProps.put(code, new ArrayList<Qproperty>(Arrays.asList(newProp)));
198               }
199             }
200           }
201           JAXBElement<Qexperiment> newDesign =
202               parser.mergeDesigns(setup, new HashSet<String>(), new ArrayList<>(), factorStructure, otherProps);
203           try {
204             String newXML = parser.toString(newDesign);
205             HashMap<String, Object> params = new HashMap<>();
206             String expID = sample.getExperimentIdentifierOrNull();
207             String[] split = expID.split("E");
208             String numSuffixOfExperiment = split[split.length - 1];
209             String infoExpID = expID.replace("E" + numSuffixOfExperiment, "_INFO");
210 
211             Map<String, Object> properties = new HashMap<>();
212             properties.put("Q_EXPERIMENTAL_SETUP", newXML);
213             params.put("user", user);
214             params.put("identifier", infoExpID);
215             params.put("properties", properties);
216             datahandler.getOpenBisClient().triggerIngestionService("update-experiment-metadata",
217                 params);
218 
219           } catch (JAXBException e) {
220             logger.error("could not update experimental design due to parser error!");
221             e.printStackTrace();
222           }
223           // try {
224           // qProperties = xmlParser.toString(xmlParser.createXMLFromProperties(factors));
225           // props.put("Q_PROPERTIES", qProperties);
226           // } catch (JAXBException e) {
227           // // TODO Auto-generated catch block
228           // e.printStackTrace();
229           // }
230         }
231 
232         HashMap<String, Object> parameters = new HashMap<String, Object>();
233         parameters.put("user", user);
234         parameters.put("identifier", id);
235         parameters.put("properties", props);
236 
237         datahandler.getOpenBisClient().triggerIngestionService("update-single-sample-metadata",
238             parameters);
239         Utils.Notification("Metadata changed succesfully",
240             String.format("Metadata values of sample %s have been commited successfully.", id),
241             "success");
242       }
243     });
244     buildFormLayout();
245     propLayout.addComponent(new Label(String.format(
246         "This view shows metadata connected to this sample and can be used to change the corresponding values. \nIdentifier: %s",
247         id), Label.CONTENT_PREFORMATTED));
248 
249     propLayout.addComponent(this.form);
250     propLayout.addComponent(saveButton);
251   }

253   private Map<String, PropertyBean> getControlledVocabularies() {
254     Map<String, PropertyBean> controlledVocabularies = new HashMap<String, PropertyBean>();
255 
256     for (PropertyType p : completeProperties) {
257       if (p instanceof ControlledVocabularyPropertyType) {
258 
259         ControlledVocabularyPropertyType controlled_vocab = (ControlledVocabularyPropertyType) p;
260         List<String> terms = new ArrayList<String>();
261 
262         for (VocabularyTerm term : controlled_vocab.getTerms()) {
263           terms.add(term.getCode().toString());
264         }
265 
266         PropertyBeanrtyBean.html#PropertyBean">PropertyBean newVocab = new PropertyBean();
267         newVocab.setCode(p.getCode());
268         newVocab.setDescription(p.getDescription());
269         newVocab.setLabel(p.getLabel());
270         newVocab.setVocabularyValues(terms);
271 
272         controlledVocabularies.put(p.getCode(), newVocab);
273       }
274     }
275     return controlledVocabularies;
276   }

279   private Map<String, PropertyBean> getProperties() {
280     Map<String, PropertyBean> properties = new HashMap<String, PropertyBean>();
281     for (PropertyType p : completeProperties) {
282       if (p.getDataType().toString().equals("XML")) {
283         continue;
284       } else if (assignedProperties.keySet().contains(p.getCode())) {
285         properties.put(p.getCode(), new PropertyBean(p.getLabel(), p.getCode(), p.getDescription(),
286             assignedProperties.get(p.getCode())));
287       } else {
288         properties.put(p.getCode(),
289             new PropertyBean(p.getLabel(), p.getCode(), p.getDescription(), ""));
290       }
291     }
292     return properties;
293 
294   }

315   private void buildFormLayout() {
316     final FieldGroup fieldGroup = new FieldGroup();
317     final FormLayout form2 = new FormLayout();
318     this.fieldGroup = new FieldGroup();
319     this.form = new FormLayout();
320 
321     Map<String, PropertyBean> controlledVocabularies = getControlledVocabularies();
322     Map<String, PropertyBean> properties = getProperties();
323     List<Property> xmlProps = getXMLProperties();
324 
325     for (Property f : xmlProps) {
326       life.qbic.xml.properties.PropertyType type = f.getType();
327 
328       String label = f.getLabel();
329       if (f.hasUnit())
330         label += " in " + f.getUnit();
331       TextField tf = new TextField(label);
332       tf.setData(type);// save property type for later, when it is written back
333       fieldGroup.bind(tf, label);
334       tf.setCaption(label);
335       tf.setDescription("Q_PROPERTIES");
336       tf.setValue((String) f.getValue());
337       form2.addComponent(tf);
338     }
339 
340     for (String key : properties.keySet()) {
341       if (controlledVocabularies.keySet().contains(key)) {
342         ComboBox select = new ComboBox(controlledVocabularies.get(key).getLabel());
343         fieldGroup.bind(select, key);
344 
345         // Add items with given item IDs
346         select.addItems(controlledVocabularies.get(key).getVocabularyValues());
347 
348         select.setDescription(controlledVocabularies.get(key).getCode());
349         select.setValue(properties.get(key).getValue());
350 
351         form2.addComponent(select);
352 
353       } else {
354         TextField tf = new TextField(key);
355         fieldGroup.bind(tf, key);
356         tf.setCaption(properties.get(key).getLabel());
357         tf.setDescription(properties.get(key).getCode());
358         tf.setValue((String) properties.get(key).getValue());
359         form2.addComponent(tf);
360       }
361     }
362     this.fieldGroup = fieldGroup;
363     this.form = form2;
364   }
