155   /**
156    * checks how many groups currently are present in raw files, and updates them
157    */
158   public void updateGroups() {
159     if (model.getRawFilesBeans().size() == 0) {
160       model.getGroups().clear();
161       return;
162     } else {
163       HashMap<Integer, List<RawFilesBean>> rawFilesSorted =
164           new HashMap<Integer, List<RawFilesBean>>();
165       for (RawFilesBean bean : model.getRawFilesBeans().getItemIds()) {
166         if (rawFilesSorted.containsKey(bean.getParameterGroup())) {
167           rawFilesSorted.get(bean.getParameterGroup()).add(bean);
168         } else {
169           ArrayList<RawFilesBean> beans = new ArrayList<RawFilesBean>();
170           beans.add(bean);
171           rawFilesSorted.put(bean.getParameterGroup(), beans);
172         }
173 
174       }
175       // remove old groups that are not used anymore
176       for (Integer key : model.getGroups().keySet()) {
177         if (!rawFilesSorted.containsKey(key)) {
178           model.getGroups().remove(key);
179         }
180       }
181       // updated groups
182       for (Integer key : rawFilesSorted.keySet()) {
183         if (model.getGroups().containsKey(key)) {
184           model.getGroups().get(key).removeFiles();
185           model.getGroups().get(key).setFiles(rawFilesSorted.get(key));
186         } else {
187           Groupodel/maxquant/Group.html#Group">Group group = new Group();
188           group.setFiles(rawFilesSorted.get(key));
189           model.getGroups().put(key, group);
190         }
191 
192       }
193     }
194     // add new groups
195     HashMap<Integer, Group> groups = new HashMap<Integer, Group>();
196     for (RawFilesBean bean : model.getRawFilesBeans().getItemIds()) {
197       if (!groups.containsKey(bean.getParameterGroup())) {
198         Groupodel/maxquant/Group.html#Group">Group group = new Group();
199         group.addFile(bean);
200         groups.put(bean.getParameterGroup(), group);
201       }
202     }
203 
204   }

375   /**
376    * creates a json object of this component. Be sure to be up to date. Otherwise, you might not get
377    * all actual groups. To be up to date execute {@link MaxQuantComponent.updateGroups}
378    * 
379    * @return
380    * @throws JSONException
381    */
382   public JSONObject toJson() throws JSONException {
383 
384     Set<Entry<Integer, Group>> entryset = model.getGroups().entrySet();
385     Iterator<Entry<Integer, Group>> iter = entryset.iterator();
386     JSONArray rawFilesArray = new JSONArray();
387     while (iter.hasNext()) {
388       Entry<Integer, Group> entry = iter.next();
389       rawFilesArray.put(entry.getValue().toJson());
390     }
391     JSONObject params = new JSONObject();
392     params.put("rawFiles", rawFilesArray);
393     params.put("fastaFiles", fastaToJson());
394     params.put("globalParams", globalParamsToJson());
395     return params;
396   }

398   /**
399    * writes all fasta related information into a JSONObject. See
400    * https://github.com/qbicsoftware/mqrun/blob/master/examples/simple.json
401    * http://mqrun.readthedocs.org/en/latest/param_format.html and
402    * https://github.com/qbicsoftware/mqrun/tree/master/mqrun/data for more information
403    * 
404    * @return
405    * @throws JSONException
406    */
407   JSONObject fastaToJson() throws JSONException {
408     JSONArray fastafiles = new JSONArray();
409     BeanItemContainer<FastaBean> selected = model.getSelectedFastaBeans();
410     for (FastaBean bean : selected.getItemIds()) {
411       // shouldn't contain file ending and it should be the file name rather than the name of the
412       // bean
413       Path p = Paths.get(bean.getPath());
414       String file = p.getFileName().toString();
415 
416       fastafiles.put(file.split("\\.")[0]);
417       // fastafiles.put(bean.getName());
418     }
419     JSONObject ret = new JSONObject();
420     ret.put("fileNames", fastafiles);
421     ret.put("firstSearch", new JSONArray());
422     return ret;
423   }

425   JSONObject globalParamsToJson() throws JSONException {
426     JSONObject glparams = new JSONObject();
427     for (Object id : globalParameters.getItemPropertyIds()) {
428       Object value = globalParameters.getItemProperty(id).getValue();
429       glparams.put((String) id,
430           (value instanceof LinkedHashSet) ? JsonHelper.fromSet((LinkedHashSet<String>) value)
431               : value);
432     }
433     glparams.put("defaults", "default");
434     return glparams;
435   }

453   /**
454    * write json to correct workflow parameter!
455    * 
456    * @throws JSONException
457    * @throws IllegalArgumentException
458    */
459   public void writeParametersToWorkflow() throws IllegalArgumentException, JSONException {
460     if (guseWorkflow instanceof GuseWorkflowRepresentation) {
461       GuseWorkflowRepresentation w = (GuseWorkflowRepresentation) guseWorkflow;
462 
463       GuseNode node = w.getNode(INITNODE_CAPTION);
464       InputPort port = node.getPort("WORKFLOW-CTD");
465 
466       // port.getParams().get("jsonParams").setValue(toJson().toString(3));
467 
468       ParameterSet updated = w.getParameters();
469       updated.getParam("MaxQuant.1.jsonParams").setValue(toJson().toString(3));
470 
471       InputList inpList = w.getData();
472 
473       // set Reference DB for 'data staging'
474       BeanItemContainer<FastaBean> selected = model.getSelectedFastaBeans();
475       for (FastaBean bean : selected.getItemIds()) {
476         inpList.getData().get("InputFiles.1.db").setValue(bean.getPath());
477       }
478 
479       // set input raw files for 'data staging'
480       // Collection<Object> selectionMulti = rawFiles.getDestination().getSelectedRows();
481       // if (selectionMulti == null || selectionMulti.isEmpty()) {
482       // showError("Warning: Nothing selected for multi input parameter " +
483       // rawFiles.getCaption());
484       // return false;
485       // Notification.show("No Input files selected!");
486       // }
487       List<String> selectedPaths = new ArrayList<String>();
488 
489       Collection<DatasetBean> selectionMulti = model.selectedDatasets();
490 
491       for (Object o : selectionMulti) {
492         DatasetBean selectedBean = (DatasetBean) o;
493 
494         try {
495           selectedPaths.add(controller.getDatasetsNfsPath(selectedBean));
496         } catch (Exception e) {
497           LOG.error(
498               "could not retrieve nfs path. Using datasetbeans getfullpath instead. "
499                   + e.getMessage(), e.getStackTrace());
500           selectedPaths.add(selectedBean.getFullPath());
501         }
502       }
503       inpList.getData().get("InputFiles.1.input").setValue(selectedPaths);
504     }
