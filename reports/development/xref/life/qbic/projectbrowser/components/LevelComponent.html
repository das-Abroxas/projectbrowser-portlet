542   private void prepareDownloads(final BeanItemContainer<DatasetBean> forExport, final String id) {
543     UI.getCurrent().setPollInterval(150);
544 
545     if (fileDownloaderData != null) {
546       this.exportData.removeExtension(fileDownloaderData);
547     }
548     final CompletableFuture<String> rawDataFuture =
549         CompletableFuture.supplyAsync(() -> Utils.containerToString(forExport));
550     final CompletableFuture<StreamResource> rawResourceFuture =
551         rawDataFuture.thenApplyAsync(rawData -> Utils.getTSVStream(rawData, String.format("%s_%s_",
552             id.substring(1).replace("/", "_"), datasetTable.getCaption().replace(" ", "_"))));
553     final CompletableFuture<Void> rawUIControlsFuture =
554         rawResourceFuture.thenAcceptAsync((stream) -> {
555           UI.getCurrent().access(() -> {
556             fileDownloaderData = new FileDownloader(stream);
557             fileDownloaderData.extend(exportData);
558           });
559         });
560 
561     if (fileDownloaderSamples != null) {
562       this.exportSamples.removeExtension(fileDownloaderSamples);
563     }
564     final CompletableFuture<String> samplesDataFuture =
565         CompletableFuture.supplyAsync(() -> Utils.containerToString(samples));
566     final CompletableFuture<StreamResource> samplesResourceFuture = samplesDataFuture
567         .thenApplyAsync(samplesData -> Utils.getTSVStream(samplesData, String.format("%s_%s_",
568             id.substring(1).replace("/", "_"), sampleGrid.getCaption().replaceAll(" ", "_"))));
569     final CompletableFuture<Void> samplesUIControlsFuture =
570         samplesResourceFuture.thenAcceptAsync((stream) -> {
571           UI.getCurrent().access(() -> {
572             fileDownloaderSamples = new FileDownloader(stream);
573             fileDownloaderSamples.extend(exportSamples);
574           });
575         });
576 
577     CompletableFuture.allOf(samplesUIControlsFuture, rawUIControlsFuture)

1077   /**
1078    * The input should have the following form: type=openbis_type&id=openbis_id e.g.
1079    * type=sample&id=/ABI_SYSBIO/QMARI117AV It is specifically designed to be used in the case of
1080    * datasetView. In other cases there is no guarantee that it will work correctly. returns a map
1081    * with two entries: "type": "openbistype" "id" : "openbisId"
1082    */
1083   public static Map<String, String> getMap(String parameters) {
1084     if (parameters == null || parameters.equals("")) {
1085       return null;
1086     }
1087     String[] params = parameters.split("&");
1088     // TODO check for length == 2 needed ?
1089     // if (params == null || params.length != 2)
1090     if (params == null || params.length > 3) {
1091       return null;
1092     }
1093     HashMap<String, String> map = new HashMap<String, String>();
1094     for (String p : params) {
1095       String[] kv = p.split("=");
1096       if (kv.length != 2) {
1097         return null;
1098       }
1099       map.put(kv[0], kv[1]);
1100     }
1101     return map;
1102   }
