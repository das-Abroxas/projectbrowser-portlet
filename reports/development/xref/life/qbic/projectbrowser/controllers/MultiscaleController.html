82    public boolean update(String id, EntityType type) {
83      jaxbelem = null;
84      String xml = null;
85      if (type.equals(EntityType.EXPERIMENT)) {
86        Experiment e = openbis.getExperimentById2(id).get(0);
87        xml = e.getProperties().get("Q_NOTES");
88        currentCode = e.getCode();
89      } else {
90        EnumSet<SampleFetchOption> fetchOptions = EnumSet.of(SampleFetchOption.PROPERTIES);
91        SearchCriteria sc = new SearchCriteria();
92        sc.addMatchClause(MatchClause.createAttributeMatch(MatchClauseAttribute.CODE, id));
93        List<Sample> samples = openbis.getOpenbisInfoService()
94            .searchForSamplesOnBehalfOfUser(openbis.getSessionToken(), sc, fetchOptions, "admin");
95        if (samples != null && samples.size() == 1) {
96          Sample sample = samples.get(0);
97          currentCode = sample.getCode();
98          xml = sample.getProperties().get("Q_NOTES");
99        }
100     }
101     try {
102       if (xml != null) {
103         jaxbelem = HistoryReader.parseNotes(xml);
104       } else {
105         jaxbelem = new JAXBElement<Notes>(new QName(""), Notes.class, new Notes());
106       }
107       currentId = id;
108       return true;
109     } catch (IndexOutOfBoundsException | JAXBException | NullPointerException e) {
110       currentId = null;
111       currentCode = null;
112       LOG.error("Error parsing XML");
113       e.printStackTrace();
114     }
115     return false;
116   }

129   public boolean addNote(Note note) {
130     if (currentId != null) {
131       jaxbelem.getValue().getNote().add(note);
132       Map<String, Object> params = new HashMap<String, Object>();
133       params.put("id", currentId);
134       params.put("user", note.getUsername());
135       params.put("comment", note.getComment());
136       params.put("time", note.getTime());
137       openbis.ingest("DSS1", "add-to-xml-note", params);
138       return true;
139     }
140     return false;
141   }

147   public List<String> getSearchResults(String samplecode) {
148     EnumSet<SampleFetchOption> fetchOptions = EnumSet.of(SampleFetchOption.PROPERTIES);
149     SearchCriteria sc = new SearchCriteria();
150     sc.addMatchClause(
151         MatchClause.createAttributeMatch(MatchClauseAttribute.CODE, samplecode + "*"));
152     List<Sample> samples = openbis.getOpenbisInfoService()
153         .searchForSamplesOnBehalfOfUser(openbis.getSessionToken(), sc, fetchOptions, user);
154     List<String> ret = new ArrayList<String>(samples.size());
155     for (Sample sample : samples) {
156       ret.add(sample.getCode());
157     }
158     return ret;
159   }

161   public String getLiferayUser(String userID) {
162     Company company = null;
163     long companyId = 1;
164     String userString = "";
165     try {
166       String webId = PropsUtil.get(PropsKeys.COMPANY_DEFAULT_WEB_ID);
167       company = CompanyLocalServiceUtil.getCompanyByWebId(webId);
168       companyId = company.getCompanyId();
169       LOG.debug(
170           String.format("Using webId %s and companyId %d to get Portal User", webId, companyId));
171     } catch (PortalException | SystemException e) {
172       LOG
173           .error("liferay error, could not retrieve companyId. Trying default companyId, which is "
174               + companyId, e.getStackTrace());
175     }
176 
177     User user = null;
178     try {
179       user = UserLocalServiceUtil.getUserByScreenName(companyId, userID);
180     } catch (PortalException | SystemException e) {
181     }
182 
183     if (user == null) {
184       LOG.warn(String.format("Openbis user %s appears to not exist in Portal", userID));
185       userString = userID;
186     } else {
187       String fullname = user.getFullName();
188       String email = user.getEmailAddress();
189       userString += ("<a href=\"mailto:");
190       userString += (email);
191       userString += ("\" style=\"color: #0068AA; text-decoration: none\">");
192       userString += (fullname);
193       userString += ("</a>");
194     }
195     return userString;
196   }
