154   void initView() {

156     setWidth(100, Unit.PERCENTAGE);

160     sampview_content.setResponsive(true);
161 
162     sampview_tab = new TabSheet();
163     sampview_tab.setResponsive(true);
164     sampview_tab.setWidth(100, Unit.PERCENTAGE);
165     sampview_tab.addStyleName(ValoTheme.TABSHEET_FRAMED);
166     sampview_tab.addStyleName(ValoTheme.TABSHEET_EQUAL_WIDTH_TABS);
167     sampview_tab.addStyleName(ValoTheme.TABSHEET_PADDED_TABBAR);
168 
169     header = "";
170 
171     datasetComponent = new DatasetComponent(datahandler, state, resourceUrl);
172     changeMetaDataComponent = new ChangeMetadataComponent(datahandler, state, resourceUrl);
173 
174     sampview_tab.addTab(initDescription()).setIcon(FontAwesome.INFO_CIRCLE);
175     // sampview_tab.addTab(initStatistics()).setIcon(FontAwesome.BAR_CHART_O);
176     sampview_tab.addTab(datasetComponent).setIcon(FontAwesome.DATABASE);
177     initNoteComponent();
178     sampview_tab.addTab(innerNotesComponent).setIcon(FontAwesome.PENCIL);
179 
180     sampview_tab.setImmediate(true);
181 
182     sampview_tab.addSelectedTabChangeListener(new SelectedTabChangeListener() {
183 
184       /**
185 		 * 
186 		 */
187       private static final long serialVersionUID = 6899763427531265769L;
188 
189       @Override
190       public void selectedTabChange(SelectedTabChangeEvent event) {
191 
192         if (event.getTabSheet().getSelectedTab().getCaption().equals("Datasets")) {
193           datasetComponent.updateUI(navigateToLabel, getCurrentBean().getId());
194         }
195       }
196     });
197 
198     sampview_content.addComponent(sampview_tab);
199     this.addComponent(sampview_content);
200   }

368   void updateContentDescription() {
369     sampleTypeLabel.setValue(String.format("Sample type: %s",
370         uglyToPretty.getPrettyName(currentBean.getType())));
371     // sampleParentLabel.setValue(currentBean.getParentsFormattedString());
372 
373     // check if it does have parents !
374     if (currentBean.getParents() == null || currentBean.getParents().isEmpty()) {
375       parentButton.setVisible(false);
376     } else if (currentBean.getParents().size() > 1) {
377       sampleParentLabel.setValue(currentBean.getParentsFormattedString());
378       parentButton.setVisible(false);
379     } else {
380       sampleParentLabel.setValue("This sample has been derived from ");
381       parentButton.setCaption(currentBean.getParents().get(0).getCode());
382 
383       parentButton.setVisible(true);
384     }
385 
386 
387     int numberOfDatasets = currentBean.getDatasets().size();
388     // numberOfDatasetsLabel.setValue(String.format("This sample has %s attached dataset(s) ",
389     // numberOfDatasets));
390     if (numberOfDatasets > 0) {
391 
392       String lastDataset = "";// "No Datasets available!";
393       if (currentBean.getLastChangedDataset() != null) {
394         lastDataset = currentBean.getLastChangedDataset().toString();
395         lastChangedDatasetLabel.setValue(String.format("Last Change: %s",
396             String.format("Dataset added on %s", lastDataset)));
397       } else {
398         lastChangedDatasetLabel.setValue(lastDataset);
399       }
400     }
401 
402     try {
403       propertiesLabel.setValue(currentBean.generatePropertiesFormattedString());
404       experimentalFactorLabel.setValue(currentBean.generateXMLPropertiesFormattedString());
405     } catch (JAXBException e) {
406       LOG.error(
407           String.format("failed to parse experimental factors for sample %s", currentBean.getId()),
408           e);
409     }
410   }

552   public void setContainerDataSource(SampleBean sampleBean) {
553     // this.currentBean = sampleBean;
554 
555     this.table.setContainerDataSource(sampleBean.getDatasets());
556     this.table.setVisibleColumns(new Object[] {"name", "type", "registrationDate", "fileSize"});
557 
558     int rowNumber = this.table.size();
559 
560     if (rowNumber == 0) {
561       this.table.setVisible(false);
562       // this.export.setVisible(false);
563       tableSectionContent.removeAllComponents();
564       tableSectionContent.addComponent(new Label("No datasets registered."));
565     } else {
566       tableSectionContent.removeAllComponents();
567       tableSectionContent.addComponent(table);
568       this.table.setVisible(true);
569       // this.export.setVisible(true);
570       this.table.setPageLength(Math.max(3, Math.min(rowNumber, 10)));
571     }
572   }

574   private FilterTreeTable buildFilterTable() {
575     FilterTreeTable filterTable = new FilterTreeTable();
576 
577     // filterTable.setSizeFull();
578 
579     filterTable.setFilterDecorator(new DatasetViewFilterDecorator());
580     filterTable.setFilterGenerator(new DatasetViewFilterGenerator());
581 
582     filterTable.setFilterBarVisible(true);
583 
584     filterTable.setSelectable(true);
585     filterTable.setImmediate(true);
586     filterTable.setMultiSelect(true);
587 
588     filterTable.setRowHeaderMode(RowHeaderMode.INDEX);
589 
590     filterTable.setColumnCollapsingAllowed(true);
591 
592     filterTable.setColumnReorderingAllowed(true);
593 
594     if (this.datasets != null) {
595       filterTable.setContainerDataSource(this.datasets);
596     }
597 
598     filterTable.setColumnHeader("name", "Name");
599     filterTable.setColumnHeader("type", "Type");
600     filterTable.setColumnHeader("registrationDate", "Registration Date");
601     filterTable.setColumnHeader("fileSize", "Size");
602 
603     return filterTable;
604   }
