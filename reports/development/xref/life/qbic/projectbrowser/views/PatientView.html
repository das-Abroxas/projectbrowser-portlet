203   /**
204    * sets the ContainerDataSource for showing it in a table and the id of the current Openbis
205    * Project. The id is shown in the caption.
206    * 
207    * @param projectBean
208    */
209   public void setContainerDataSource(ProjectBean projectBean) {
210     this.currentBean = projectBean;
211 
212     registeredExperiments.setContainerDataSource(projectBean.getExperiments());
213 
214     registeredExperiments.setVisibleColumns(
215         new Object[] {"code", "prettyType", "registrationDate", "registrator", "status"});
216 
217     registeredExperiments.setColumnHeader("prettyType", "Type");
218 
219     int rowNumber = projectBean.getExperiments().size();
220 
221     if (rowNumber == 0) {
222       registeredExperiments.setVisible(false);
223     } else {
224       registeredExperiments.setVisible(true);
225       registeredExperiments.setPageLength(Math.min(rowNumber, 10));
226     }
227   }

295       @Override
296       public void selectedTabChange(SelectedTabChangeEvent event) {
297         TabSheet tab = (TabSheet) event.getSource();
298 
299         if (event.getTabSheet().getSelectedTab().getCaption().equals("Project Graph")) {
300           loadGraph();
301         } else if (event.getTabSheet().getSelectedTab().getCaption().equals("Exp. Steps")) {
302           experimentComponent.updateUI(getCurrentBean());
303         } else if (event.getTabSheet().getSelectedTab().getCaption().equals("Datasets")) {
304           datasetComponent.updateUI("project", getCurrentBean().getId());
305         } else if (event.getTabSheet().getSelectedTab().getCaption().equals("Raw Data")) {
306           measuredSamplesComponent.updateUI("project", getCurrentBean().getId(), "measured");
307         } else if (event.getTabSheet().getSelectedTab().getCaption().equals("Biological Samples")) {
308           biologicalSamplesComponent.updateUI(getCurrentBean().getId());
309         } else if (event.getTabSheet().getSelectedTab().getCaption().equals("Results")) {
310           resultsComponent.updateUI("project", getCurrentBean().getId(), "results");
311         } else if (event.getTabSheet().getSelectedTab().getCaption().equals("Status")) {
312           statusComponent.updateUI(getCurrentBean());
313         } else if (event.getTabSheet().getSelectedTab().getCaption().equals("Workflows")) {
314           Map<String, String> args = new HashMap<String, String>();
315           args.put("id", getCurrentBean().getId());
316           args.put("type", "project");
317           workflowComponent.update(args);
318         } else if (event.getTabSheet().getSelectedTab().getCaption().equals("Upload Files")) {
319           // (get space from currentBean)
320           uploadComponent.updateUI(manager, getCurrentBean().getCode(),
321               currentBean.getId().split("/")[1], datahandler.getOpenBisClient());
322         } else if (event.getTabSheet().getSelectedTab().getCaption().equals("")) {
323           projectInformation.updateUI(getCurrentBean(), "patient");
324         }
325       }
326     });
327 
328     patientViewContent.addComponent(patientViewTab);
329     this.addComponent(patientViewContent);
330 
331   }

518   void updateHLALayout() {
519 
520     String labelContent = "<head> <title></title> </head> <body> ";
521 
522     Boolean available = false;
523 
524     SearchCriteria sc = new SearchCriteria();
525     sc.addMatchClause(MatchClause.createAttributeMatch(MatchClauseAttribute.TYPE,
526         ExperimentType.Q_NGS_HLATYPING.name()));
527     SearchCriteria projectSc = new SearchCriteria();
528     projectSc.addMatchClause(
529         MatchClause.createAttributeMatch(MatchClauseAttribute.PROJECT, currentBean.getCode()));
530     sc.addSubCriteria(SearchSubCriteria.createExperimentCriteria(projectSc));
531 
532     SearchCriteria experimentSc = new SearchCriteria();
533     experimentSc.addMatchClause(MatchClause.createAttributeMatch(MatchClauseAttribute.TYPE,
534         ExperimentType.Q_NGS_HLATYPING.name()));
535     sc.addSubCriteria(SearchSubCriteria.createExperimentCriteria(experimentSc));
536 
537 
538     List<Sample> samples = datahandler.getOpenBisClient().getFacade().searchForSamples(sc);
539 
540     SearchCriteria sc2 = new SearchCriteria();
541     sc2.addMatchClause(MatchClause.createAttributeMatch(MatchClauseAttribute.TYPE,
542         ExperimentType.Q_WF_NGS_HLATYPING.name()));
543     SearchCriteria projectSc2 = new SearchCriteria();
544     projectSc2.addMatchClause(
545         MatchClause.createAttributeMatch(MatchClauseAttribute.PROJECT, currentBean.getCode()));
546     sc2.addSubCriteria(SearchSubCriteria.createExperimentCriteria(projectSc2));
547 
548     SearchCriteria experimentSc2 = new SearchCriteria();
549     experimentSc2.addMatchClause(MatchClause.createAttributeMatch(MatchClauseAttribute.TYPE,
550         ExperimentType.Q_WF_NGS_HLATYPING.name()));
551     sc2.addSubCriteria(SearchSubCriteria.createExperimentCriteria(experimentSc2));
552 
553     List<Experiment> wfExperiments =
554         datahandler.getOpenBisClient().getFacade().searchForExperiments(sc2);
555 
556     List<Sample> wfSamples = new ArrayList<Sample>();
557 
558     for (Experiment exp : wfExperiments) {
559       if (exp.getCode().contains(currentBean.getCode())) {
560         wfSamples
561             .addAll(datahandler.getOpenBisClient().getSamplesofExperiment(exp.getIdentifier()));
562       }
563     }
564 
565     for (Sample sample : samples) {
566       available = true;
567       String classString = sample.getProperties().get("Q_HLA_CLASS");
568       String[] splitted = classString.split("_");
569       String lastOne = splitted[splitted.length - 1];
570       String addInformation = "";
571 
572       if (!(sample.getProperties().get("Q_ADDITIONAL_INFO") == null)) {
573         addInformation = sample.getProperties().get("Q_ADDITIONAL_INFO");
574       }
575 
576       labelContent += String.format("MHC Class %s " + "<p><u>Patient</u>: %s </p> " + "<p>%s </p> ",
577           lastOne, sample.getProperties().get("Q_HLA_TYPING"), addInformation);
578     }
579 
580     for (Sample sample : wfSamples) {
581       available = true;
582       labelContent += String.format("<u>Computational Typing (OptiType)</u>" + "<p> %s </p> ",
583           sample.getProperties().get("Q_HLA_TYPING"));
584     }
585 
586     labelContent += "</body>";
587     if (available) {
588       hlaTypeLabel.setValue(labelContent);
589     }
590 
591     else {
592       hlaTypeLabel.setValue("Not available.");
593     }
594   }
