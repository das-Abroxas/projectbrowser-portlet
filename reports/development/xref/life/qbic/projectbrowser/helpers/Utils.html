76      /**
77       * Checks if a String can be parsed to an Integer
78       *
79       * @param s a String
80       * @return true, if the String can be parsed to an Integer successfully, false otherwise
81       */
82      public static boolean isInteger(String s) {
83          try {
84              Integer.parseInt(s);
85          } catch (NumberFormatException e) {
86              return false;
87          }
88          return true;
89      }

91      /**
92       * Parses a whole String list to integers and returns them in another list.
93       *
94       * @param strings List of Strings
95       * @return list of integer representations of the input list
96       */
97      public static List<Integer> strArrToInt(List<String> strings) {
98          List<Integer> res = new ArrayList<Integer>();
99          for (String s : strings) {
100             res.add(Integer.parseInt(s));
101         }
102         return res;
103     }

105     /**
106      * Maps an integer to a char representation. This can be used for computing the checksum.
107      *
108      * @param i number to be mapped
109      * @return char representing the input number
110      */
111     public static char mapToChar(int i) {
112         i += 48;
113         if (i > 57) {
114             i += 7;
115         }
116         return (char) i;
117     }

119     /**
120      * Checks which of two Strings can be parsed to a larger Integer and returns it.
121      *
122      * @param a a String
123      * @param b another String
124      * @return the String that represents the larger number.
125      */
126     public static String max(String a, String b) {
127         int a1 = Integer.parseInt(a);
128         int b1 = Integer.parseInt(b);
129         if (Math.max(a1, b1) == a1) {
130             return a;
131         } else {
132             return b;
133         }
134     }

136     /**
137      * Creates a string with leading zeroes from a number
138      *
139      * @param id number
140      * @param length of the final string
141      * @return the completed String with leading zeroes
142      */
143     public static String createCountString(int id, int length) {
144         String res = Integer.toString(id);
145         while (res.length() < length) {
146             res = "0" + res;
147         }
148         return res;
149     }

151     /**
152      * Increments the value of an upper case char. When at "X" restarts with "A".
153      *
154      * @param c the char to be incremented
155      * @return the next letter in the alphabet relative to the input char
156      */
157     public static char incrementUppercase(char c) {
158         if (c == 'X') {
159             return 'A';
160         } else {
161             int charValue = c;
162             return (char) (charValue + 1);
163         }
164     }

191     // TODO fix and test
192     public static String containerToString(final BeanItemContainer container) {
193         final StringBuilder stringValue = new StringBuilder();
194         final Set<String> exclusionList = new HashSet<>(Arrays
195             .asList("experiments", "dssPath", "children", "samples", "properties", "controlledVocabularies", "typeLabels", "containsData", "parent", "parents",
196                 "isSelected", "root"));
197 
198         final Collection<String> filteredPropertyIDs = ((Collection<String>) container.getContainerPropertyIds()).parallelStream()
199             .filter(p -> !exclusionList.contains(p)).collect(Collectors.toList());
200 
201         for (final Object o : filteredPropertyIDs) {
202             final String propertyAsString = o.toString();
203             stringValue.append(propertyAsString.replaceAll("project", "sub-project").replace("space", "project")).append('\t');
204         }
205         stringValue.append('\n');
206 
207         for (final Object id : container.getItemIds()) {
208             final Item item = container.getItem(id);
209 
210             for (final Object propertyId : filteredPropertyIDs) {
211                 // Could be extended to an exclusion list if we don't want to show further columns
212                 final Property prop = item.getItemProperty(propertyId);
213                 if (prop.getValue() == null) {
214                     stringValue.append("-\t");
215                 } else {
216                     stringValue.append(prop.toString()).append('\t');
217                 }
218             }
219             stringValue.append('\n');
220         }
221         return stringValue.toString();
222     }

274     public static void Notification(String title, String description, String type) {
275         Notification notify = new Notification(title, description);
276         notify.setPosition(Position.TOP_CENTER);
277         if (type.equals("error")) {
278             notify.setDelayMsec(16000);
279             notify.setIcon(FontAwesome.FROWN_O);
280             notify.setStyleName(ValoTheme.NOTIFICATION_ERROR + " " + ValoTheme.NOTIFICATION_CLOSABLE);
281         } else if (type.equals("success")) {
282             notify.setDelayMsec(8000);
283             notify.setIcon(FontAwesome.SMILE_O);
284             notify.setStyleName(ValoTheme.NOTIFICATION_SUCCESS + " " + ValoTheme.NOTIFICATION_CLOSABLE);
285         } else {
286             notify.setDelayMsec(8000);
287             notify.setIcon(FontAwesome.COMMENT);
288             notify.setStyleName(ValoTheme.NOTIFICATION_TRAY + " " + ValoTheme.NOTIFICATION_CLOSABLE);
289         }
290         notify.show(Page.getCurrent());
291     }

306     public static String usernameToFullName(String username) {
307         Company company = null;
308 
309         String res = username;
310         long companyId = 1;
311         try {
312             String webId = PropsUtil.get(PropsKeys.COMPANY_DEFAULT_WEB_ID);
313             company = CompanyLocalServiceUtil.getCompanyByWebId(webId);
314             companyId = company.getCompanyId();
315         } catch (PortalException | SystemException e) {
316             LOG
317                 .error("liferay error, could not retrieve companyId. Trying default companyId, which is "
318                     + companyId, e.getStackTrace());
319         }
320         User user = null;
321         try {
322             user = UserLocalServiceUtil.getUserByScreenName(companyId, username);
323         } catch (PortalException | SystemException e) {
324             LOG.warn("got this error while trying to fetch full name of user:");
325             LOG.warn(e.getMessage());
326             LOG.info("returning username instead.");
327         }
328         if (user == null) {
329             LOG.warn(String.format("Openbis user %s appears to not exist in Portal", username));
330         } else {
331             String firstName = user.getFirstName();
332             String lastName = user.getLastName();
333             res = firstName + " " + lastName;
334         }
335         return res;
336     }
